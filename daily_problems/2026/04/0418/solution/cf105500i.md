**提示 1：** 考虑将数字从小到大插入数组。

看提示。

最小的元素安排完了之后，中间有 $x$ 个位置需要填入比它大 $1$ 的元素，设有 $y$ 个，则填入的方案数等于将 $y$ 分为 $x$ 个非空组的方案数，乘以 $y$ 个元素的排列。前者可以用隔板法算出来方案数。

而此时，新的需要填入元素的位置变为 $y-x$ ，我们再填入比这个元素大 $1$ 的元素即可。

最后，遍历完全部元素时，需要剩余的填入位置的数量为 $0$ ，否则一定无法构造。

最后，我们可以旋转整个数组得到新的数组，因此方案数除以最小值数量再乘以 $n$ 。因为我们一开始的计算方式只考虑了最小值得是第一项，所以剩余的非最小值开头的还得算进来，总数量和最小值在开头的数组的比例为 $n:x$ 。

时间复杂度为 $\mathcal{O}(n+M)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    nums = LII()
    
    M = 10 ** 6 + 5
    mod = 998244353
    
    f = Factorial(n, mod)
    
    cnt = [0] * M
    for x in nums:
        cnt[x] += 1
    
    mi = min(nums)
    ma = max(nums)
    
    ans = f.fac(cnt[mi])
    neighbors = cnt[mi]
    
    for i in range(mi + 1, ma + 1):
        if cnt[i] < neighbors:
            ans = 0
            break
        
        ans = ans * f.fac(cnt[i]) % mod
        ans = ans * f.combi(cnt[i] - 1, neighbors - 1) % mod
        
        neighbors = cnt[i] - neighbors
    
    if neighbors == 0: print(ans * n % mod * f.inv(cnt[mi]) % mod)
    else: print(0)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	int M = 1e6 + 5, mod = 998244353;

	vector<int> f(n + 1), g(n + 1);
	f[0] = 1;
	for (int i = 1; i <= n; i ++) f[i] = 1ll * i * f[i - 1] % mod;
	g[n] = quickPow(f[n], mod - 2, mod);
	for (int i = n; i >= 1; i --) g[i - 1] = 1ll * i * g[i] % mod;

	vector<int> cnt(M, 0);
	int mi = M, ma = 0;

	for (int i = 0; i < n; i ++) {
		int x;
		cin >> x;
		mi = min(x, mi);
		ma = max(x, ma);
		cnt[x] ++;
	}

	int ans = f[cnt[mi]], neighbors = cnt[mi];

	for (int i = mi + 1; i <= ma; i ++) {
		if (cnt[i] < neighbors) {
			ans = 0;
			break;
		}

		ans = 1ll * ans * f[cnt[i]] % mod * f[cnt[i] - 1] % mod * g[neighbors - 1] % mod * g[cnt[i] - neighbors] % mod;

		neighbors = cnt[i] - neighbors;
	}

	if (!neighbors) cout << 1ll * ans * n % mod * f[cnt[mi] - 1] % mod * g[cnt[mi]] % mod;
	else cout << 0;

	return 0;
}
```
