**提示 1：** 首先不能用二分，那怎么办？

**提示 2：** 扫一遍，过程中怎么保证总次数不超标？

先看提示 1，怎么办呢？

我们先暴力找第一列中所有的边界上的点。我们看如何借此找到第二列的点。

我们先考虑如何确定下边界。

整体的逻辑是下边界先往下走，再往上走。

于是我们直接根据前一列的边界，直接向右平移。

- 如果在图形内，则继续往下走，直到找到这一层的边界。

- 否则，边界一定在上面。为什么呢？不然的话，设此时横坐标是 $x$ ，两条线段的斜率都小于 $-2$ ；此时考虑这条边界的上下边界 $l,r$ ，有 $l\leq t-2(x-1),r\geq b+2(n-x)$ ，于是 $r-l\geq b-t+2(n-1)\geq 0-(n-1)+2(n-1)=n-1$ ，只可能刚好都取等号，而等号的情况下 $l=1,r=n$ ，不存在上述不成立的情况，所以不可能边界不在上面。于是暴力往上找到边界就行。

- 注意移动过程中边界不能越过起点和终点的线段，这样才能避免在区间长度为 $0$ 时跳过。（下面代码这里有问题）

这样总操作次数最多就是往下走加上往上走，是 $\mathcal{O}(n)$ 的。

而上边界也是一样的逻辑。

注意最后输出答案的时候不要换行，不然这题会判你错。

查询次数是 $\mathcal{O}(n)$ 的。

Python 做法如下——

```Python []
def main(): 
    d = {
        'X': 0,
        'P': 1,
        'I': 2
    }
    
    def query(x, y):
        print('?', x, y, flush=True)
        return d[I()]
    
    n, t, b = MII()
    
    ans = [[0] * (n + 1) for _ in range(n + 1)]
    l, r = t, t
    
    for i in range(1, n + 1):
        ans[1][i] = query(1, i)
        ans[n][i] = query(n, i)
        if ans[1][i]:
            l = fmin(l, i)
            r = fmax(r, i)
    
    for i in range(2, n):
        ans[i][l] = query(i, l)
        
        if ans[i][l]:
            while l > 1:
                l -= 1
                ans[i][l] = query(i, l)
                if ans[i][l] == 0 or l > lbound:
                    l += 1
                    break
        else:
			lbound = (b - l) // (n - (i - 1)) + l
            while l <= lbound:
                l += 1
                ans[i][l] = query(i, l)
                if ans[i][l]:
                    break
        
        ans[i][r] = query(i, r)
        
        if ans[i][r]:
            while r < n:
                r += 1
                ans[i][r] = query(i, r)
                if ans[i][r] == 0:
                    r -= 1
                    break
        else:
			rbound = (b - r - 1) // (n - (i - 1)) + 1 + l
            while r > 1:
                r -= 1
                ans[i][r] = query(i, r)
                if ans[i][r]:
                    break
        
        for j in range(l + 1, r):
            ans[i][j] = 2
    
    print('!', sum(sum(x) for x in ans) - 2, end='')
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	auto query = [&] (int i, int j) -> int {
		cout << "? " << i << ' ' << j << endl;
		char c;
		cin >> c;
		if (c == 'X') return 0;
		if (c == 'P') return 1;
		return 2;
	};

	int n, t, b;
	cin >> n >> t >> b;

	vector<vector<int>> ans(n + 1, vector<int>(n + 1, 0));
	int l = t, r = t;

	for (int i = 1; i <= n; i ++) {
		ans[1][i] = query(1, i);
		ans[n][i] = query(n, i);

		if (ans[1][i]) {
			l = min(l, i);
			r = max(r, i);
		}
	}

	for (int i = 2; i < n; i ++) {
		ans[i][l] = query(i, l);

		if (ans[i][l]) {
			while (l > 1) {
				l --;
				ans[i][l] = query(i, l);
				if (!ans[i][l]) {
					l ++;
					break;
				}
			}
		}
		else {
			while (l < n) {
				l ++;
				ans[i][l] = query(i, l);
				if (ans[i][l]) break;
			}
		}

		ans[i][r] = query(i, r);

		if (ans[i][r]) {
			while (r < n) {
				r ++;
				ans[i][r] = query(i, r);
				if (!ans[i][r]) {
					r --;
					break;
				}
			}
		}
		else {
			while (r > 1) {
				r --;
				ans[i][r] = query(i, r);
				if (ans[i][r]) break;
			}
		}

		for (int j = l + 1; j < r; j ++) ans[i][j] = 2;
	}

	int area = 0;
	for (auto &x: ans) for (auto &y: x) area += y;

	cout << "! " << area - 2;

	return 0;
}
```
