**提示 1：** 如果我们定下来最长边是什么了之后，剩余边就好办了。

**提示 2：** 选一个最长边。

假设我们定下来最长边了，那么剩余的只需要不超过这个最长边，且越长越好就行。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    ls = LII()
    rs = LII()
    
    idx = ls.index(max(ls))
    ans = [0] * n
    
    total = 0
    for i in range(n):
        if i != idx:
            ans[i] = fmin(rs[i], ls[idx])
            total += ans[i]
        else: ans[i] = ls[i]
    
    if total >= ls[idx]: print(*ans)
    else: print(-1)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	vector<int> ls(n), rs(n);
	for (auto &x: ls) cin >> x;
	for (auto &x: rs) cin >> x;

	int idx = max_element(ls.begin(), ls.end()) - ls.begin();
	long long total = 0;

	for (int i = 0; i < n; i ++) {
		if (i != idx) {
			total += min(rs[i], ls[idx]);
		}
	}

	if (total < ls[idx]) cout << -1;
	else {
		for (int i = 0; i < n; i ++) {
			cout << min(rs[i], ls[idx]) << "\n "[i < n - 1];
		}
	}

	return 0;
}
```
