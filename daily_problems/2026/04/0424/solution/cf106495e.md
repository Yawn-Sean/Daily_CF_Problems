**提示 1：** 按照质数，依次考虑。

考虑将质因数分解看成若干段，每一段是相同的质数。

对于某个非 $p$ 的数 $x$ ，在乘以某个质数的 $p^a$ 时，对应的字典序排列是 $x,xp,xp^2,\dots$ 。

而如果还有另外的质数，则这一段序列应该倒序排，这样可以让下一个质数出现的位置往后靠。

所以可以使用 DFS ，在需要考虑之后的质数的情况下，按照降序 DFS；如果不加后面的质数，则按照升序进行答案的输出。DFS 的顺序依次考虑了各个质因数和次数，因此 DFS 得到的结果就是序列排序结果。

最后查询数组输出即可。

如果你用线性筛搭配 DFS ，则时间复杂度为 $\mathcal{O}(n+q)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, q = MII()
    
    pr = [1] * (n + 1)
    primes = []
    
    for i in range(2, n + 1):
        if pr[i]:
            primes.append(i)
            for j in range(i * 2, n + 1, i):
                pr[j] = 0
    
    ans = [1]
    stk = [(1, 0)]
    
    while stk:
        x, idx = stk.pop()
        if idx == len(primes) or x * primes[idx] > n: continue
        
        stk.append((x, idx + 1))
        while x * primes[idx] <= n:
            x *= primes[idx]
            ans.append(x)
            stk.append((x, idx + 1))
    
    outs = []
    for _ in range(q):
        idx = II()
        outs.append(ans[idx - 1])
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, q;
	cin >> n >> q;

	vector<int> is_prime(n + 1, 1), primes;

	for (int i = 2; i <= n; i ++) {
		if (is_prime[i]) {
			primes.emplace_back(i);
			for (int j = i * 2; j <= n; j += i) {
				is_prime[j] = 0;
			}
		}
	}

	vector<int> ans = {1};
	vector<pair<int, int>> stk;
	stk.emplace_back(1, 0);

	while (!stk.empty()) {
		auto [x, idx] = stk.back(); stk.pop_back();
		if (idx == primes.size() || x * primes[idx] > n) continue;

		stk.emplace_back(x, idx + 1);

		while (1ll * x * primes[idx] <= n) {
			x *= primes[idx];
			ans.emplace_back(x);
			stk.emplace_back(x, idx + 1);
		}
	}

	while (q --) {
		int idx; cin >> idx;
		cout << ans[idx - 1] << '\n';
	}

	return 0;
}
```
