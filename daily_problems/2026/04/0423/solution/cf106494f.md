**提示 1：** 不修改的话这题咋做？

**提示 2：** 修改影响了哪些部分？可以更新吗？

如果不修改，我们可以直接用 $dp[(l,r)]$ 表示 $[l,r]$ 最后对应区间的期望长度。

如果 $[l,r]$ 排好序了，则答案是 $r-l+1$ ，否则等于分成的两个子数组对应的答案的平均值。

这里涉及到的区间都是区间不断分成 $2$ 个得到的区间，因此总区间数量是 $\mathcal{O}(n)$ 的，类似线段树中节点个数的推导。

所以我们可以在这个类似线段树的结构中，从叶子节点不断往根节点更新上述 DP。同时，DP 转移过程中需要判断每一段是否已经排好序了，所以也需要多维护一个表示这件事的变量。

接下来考虑更新。更新的时候，相当于更新了所有包含 $idx$ 位置的区间，也就是一条从叶子到根的路径，判断这些区间是否排好序了，同时更新对应 DP 结果。

怎么快速判断区间是否排序呢？我们直接统计不满足排序关系的相邻对的数量即可，这样从叶子到根节点的数量维护只需要对左右儿子求和，再加上相接部分是否满足排序要求。而 DP 的数值可以根据这件事更新。

最后输出根节点的 DP 结果即可。

时间复杂度为 $\mathcal{O}(n+q\log n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, q = MII()
    nums = LII()
    
    mod = 998244353
    rev2 = (mod + 1) // 2
    
    M = 2200000
    seg_tree = [0] * M
    dp = [1] * M
    
    ls = [-1] * M
    rs = [-1] * M
    pos = [0] * n
    
    ls[1] = 0
    rs[1] = n - 1
    
    for i in range(1, M):
        if rs[i] > ls[i]:
            mid = (ls[i] + rs[i]) // 2
            ls[2 * i] = ls[i]
            rs[2 * i] = mid
            ls[2 * i + 1] = mid + 1
            rs[2 * i + 1] = rs[i]
        elif ls[i] >= 0:
            pos[ls[i]] = i
    
    for i in range(M - 1, -1, -1):
        if ls[i] < rs[i]:
            mid = (ls[i] + rs[i]) // 2
            seg_tree[i] = seg_tree[2 * i] + seg_tree[2 * i + 1] + (nums[mid] > nums[mid + 1])
            if seg_tree[i] == 0: dp[i] = rs[i] - ls[i] + 1
            else: dp[i] = (dp[2 * i] + dp[2 * i + 1]) * rev2 % mod
    
    outs = []
    for _ in range(q):
        idx, val = MII()
        idx -= 1
        
        nums[idx] = val
        p = pos[idx] // 2
        
        while p:
            mid = (ls[p] + rs[p]) // 2
            seg_tree[p] = seg_tree[2 * p] + seg_tree[2 * p + 1] + (nums[mid] > nums[mid + 1])
            if seg_tree[p] == 0: dp[p] = rs[p] - ls[p] + 1
            else: dp[p] = (dp[2 * p] + dp[2 * p + 1]) * rev2 % mod
            p //= 2
        
        outs.append(dp[1])
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, q, mod = 998244353, rev2 = (mod + 1) / 2;
	cin >> n >> q;

	vector<int> nums(n);
	for (auto &x: nums) cin >> x;

	int M = 2'200'000;
	vector<int> seg_tree(M, 0), dp(M, 1), ls(M, -1), rs(M, -1), pos(n);

	ls[1] = 0;
	rs[1] = n - 1;

	for (int i = 1; i < M; i ++) {
		if (rs[i] > ls[i]) {
			int mid = (ls[i] + rs[i]) / 2;
			ls[2 * i] = ls[i];
			rs[2 * i] = mid;
			ls[2 * i + 1] = mid + 1;
			rs[2 * i + 1] = rs[i];
		}
		else if (ls[i] >= 0) pos[ls[i]] = i;
	}

	for (int i = M - 1; i >= 0; i --) {
		if (ls[i] < rs[i]) {
			int mid = (ls[i] + rs[i]) / 2;
			seg_tree[i] = seg_tree[2 * i] + seg_tree[2 * i + 1] + (nums[mid] > nums[mid + 1]);
			if (!seg_tree[i]) dp[i] = rs[i] - ls[i] + 1;
			else dp[i] = 1ll * (dp[2 * i] + dp[2 * i + 1]) * rev2 % mod;
		}
	}

	while (q --) {
		int idx, val;
		cin >> idx >> val; idx --;

		nums[idx] = val;
		int p = pos[idx] / 2;

		while (p) {
			int mid = (ls[p] + rs[p]) / 2;
			seg_tree[p] = seg_tree[2 * p] + seg_tree[2 * p + 1] + (nums[mid] > nums[mid + 1]);
			if (!seg_tree[p]) dp[p] = rs[p] - ls[p] + 1;
			else dp[p] = 1ll * (dp[2 * p] + dp[2 * p + 1]) * rev2 % mod;
			p /= 2;
		}

		cout << dp[1] << '\n';
	}

	return 0;
}
```
