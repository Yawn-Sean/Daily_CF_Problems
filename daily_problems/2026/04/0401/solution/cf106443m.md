**提示 1：** 更新的是什么东西？如何更新？怎么写更方便？

对于每个人的位置，我们在数组的其他位置加上这个人到该位置的距离。

这样我们更新的就是以这个点为中心的等差数列，同时更新方式是做加法。于是用差分更新。

注意这里更新的相当于是一条线段，所以我们直接理解为加一个一次函数。于是用一个差分数组表示一次项系数，一个差分数组表示常数系数，这样差分的逻辑就比较容易写对了。

同时这里环形数组，容易涉及越界问题。可以考虑复制一遍数组，断环为链，这样就能在连续的一段区间上操作了，最后再把实际相同位置的结果加起来就行。

最后取更新结果的最小值。

时间复杂度为 $\mathcal{O}(n+m)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    n, m = MII()
    nums = LII()
    
    diff_k = [0] * (m * 2 + 1)
    diff_b = [0] * (m * 2 + 1)
    
    v = (m + 1) // 2
    
    for x in nums:
        diff_k[x] += 1
        diff_k[x + v] -= 1
        diff_b[x] -= x
        diff_b[x + v] += x
        
        diff_k[x + v] -= 1
        diff_k[x + m] += 1
        diff_b[x + v] += x + m
        diff_b[x + m] -= x + m
    
    for i in range(m * 2):
        diff_k[i + 1] += diff_k[i]
        diff_b[i + 1] += diff_b[i]
    
    ans = [0] * m
    
    for i in range(m * 2):
        ans[i % m] += diff_k[i] * i + diff_b[i]
    
    print(min(ans))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m;
	cin >> n >> m;

	vector<long long> diff_k(2 * m + 1, 0), diff_b(2 * m + 1, 0);
	int v = (m + 1) / 2;

	while (n --) {
		int x;
		cin >> x;

		diff_k[x] += 1;
		diff_k[x + v] -= 1;
		diff_b[x] -= x;
		diff_b[x + v] += x;

		diff_k[x + v] -= 1;
		diff_k[x + m] += 1;
		diff_b[x + v] += x + m;
		diff_b[x + m] -= x + m;
	}

	for (int i = 0; i < 2 * m; i ++) {
		diff_k[i + 1] += diff_k[i];
		diff_b[i + 1] += diff_b[i];
	}

	vector<long long> ans(m, 0);
	for (int i = 0; i < 2 * m; i ++)
		ans[i % m] += diff_k[i] * i + diff_b[i];

	cout << *min_element(ans.begin(), ans.end());

	return 0;
}
```
