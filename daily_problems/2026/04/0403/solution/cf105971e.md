**提示 1：** 每一个空缺步骤的可选集很小。

**提示 2：** 考虑成图论问题的话，是什么问题？

首先，给出的步骤之间只隔了两步，同时不会有重复，所以曼哈顿距离一定是 $2$ 。一旦有不满足的直接就输出 $0$ 。

否则，可选点集大小不超过 $2$ 。我们将两个点相连形成一条边（如果只有一个点就连自环）。

我们在这个图中，每条边钦定一个端点，使得端点的选择不重合。

考虑每个连通块。最后答案方案数就是每个连通块的答案的乘积。

如果连通块中不止包含一个环，则边数大于点数，直接点不够用了，矛盾。

如果只有一个环——如果是自环，则这个环只有一种选法，剩余的点都确定了，所以只有一种方案；否则，环大小不小于 $2$ ，环上的分配点方案有 $2$ 种，外侧的边只有一种分配方式。

如果没有环，也就是是一棵树，此时可以任选一点为空，其余的分配也就确定了。所以方案数等于点数。

时间复杂度为 $\mathcal{O}(nm+k)$ ，如果你用 BFS / DFS 来处理中间的部分，当然用并查集实现似乎更好写。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    t = II()
    outs = []
    
    mod = 10 ** 9 + 7
    
    for _ in range(t):
        n, m, k = MII()
        xs = []
        ys = []
        
        for _ in range(k + 1):
            x, y = GMI()
            xs.append(x)
            ys.append(y)
        
        flg = True
        for i in range(k):
            if abs(xs[i + 1] - xs[i]) + abs(ys[i + 1] - ys[i]) != 2:
                flg = False
        
        if not flg: outs.append(0)
        else:
            uf = UnionFind(n * m)
            edge_cnt = [0] * (n * m)
            self_cycle = [0] * (n * m)
            
            def f(x, y):
                return x * m + y
            
            for i in range(k):
                x1, y1 = xs[i], ys[i]
                x2, y2 = xs[i + 1], ys[i + 1]
    
                if x1 == x2 or y1 == y2:
                    msk = f((x1 + x2) // 2, (y1 + y2) // 2)
                    edge_cnt[msk] += 1
                    self_cycle[msk] = 1
                else:
                    msk1 = f(x1, y2)
                    msk2 = f(x2, y1)
                    uf.merge(msk1, msk2)
                    edge_cnt[msk1] += 1
            
            for i in range(n * m):
                if uf.find(i) != i:
                    edge_cnt[uf.find(i)] += edge_cnt[i]
                    self_cycle[uf.find(i)] |= self_cycle[i]
            
            ans = 1
            for i in range(n * m):
                if uf.find(i) == i:
                    if edge_cnt[i] > uf.getSize(i):
                        ans = 0
                    elif edge_cnt[i] == uf.getSize(i):
                        if not self_cycle[i]:
                            ans = ans * 2 % mod
                    else:
                        ans = ans * uf.getSize(i) % mod
            
            outs.append(ans)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t, mod = 1e9 + 7;
	cin >> t;

	while (t --) {
		int n, m, k;
		cin >> n >> m >> k;

		vector<pair<int, int>> pts(k + 1);
		for (auto &[x, y]: pts) {
			cin >> x >> y;
			x --, y --;
		}

		bool flg = true;
		for (int i = 0; i < k; i ++) {
			auto [x1, y1] = pts[i];
			auto [x2, y2] = pts[i + 1];
			if (abs(x1 - x2) + abs(y1 - y2) != 2) flg = false;
		}

		if (!flg) cout << 0 << '\n';
		else {
			atcoder::dsu uf(n * m);
			vector<int> edge_cnt(n * m, 0), self_cycle(n * m, 0);

			auto f = [&] (int x, int y) -> int {return x * m + y;};

			for (int i = 0; i < k; i ++) {
				auto [x1, y1] = pts[i];
				auto [x2, y2] = pts[i + 1];
				
				if (x1 == x2 || y1 == y2) {
					int msk = f((x1 + x2) / 2, (y1 + y2) / 2);
					edge_cnt[msk] ++;
					self_cycle[msk] = 1;
				}
				else {
					int msk1 = f(x1, y2);
					int msk2 = f(x2, y1);
					uf.merge(msk1, msk2);
					edge_cnt[msk1] ++;
				}
			}

			for (int i = 0; i < n * m; i ++) {
				if (uf.leader(i) != i) {
					edge_cnt[uf.leader(i)] += edge_cnt[i];
					self_cycle[uf.leader(i)] |= self_cycle[i];
				}
			}

			int ans = 1;
			for (int i = 0; i < n * m; i ++) {
				if (uf.leader(i) == i) {
					if (edge_cnt[i] > uf.size(i)) ans = 0;
					else if (edge_cnt[i] == uf.size(i)) {
						if (!self_cycle[i]) ans = 2 * ans % mod;
					}
					else ans = 1ll * uf.size(i) * ans % mod;
				}
			}

			cout << ans << '\n';
		}
	}

	return 0;
}
```
