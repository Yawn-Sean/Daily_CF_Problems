**提示 1：** 可以打印一下打乱的过程。

当你将 $[1,2,\dots,52]$ 进行打乱并输出结果后，你会发现结果是 $8$ 个一循环的。

因此只需模拟 $8$ 次看是否有满足要求的即可。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    nums = LII()
    
    for _ in range(8):
        nnums = []
        for i in range(26):
            nnums.append(nums[i])
            nnums.append(nums[i + 26])
        nums = nnums
        
        flg = True
        for i in range(26, 52):
            if nums[i] == 1:
                flg = False
        
        if flg: exit(print('YES'))
    
    print('NO')
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	vector<int> nums(52);
	for (auto &v: nums) cin >> v;

	for (int i = 0; i < 8; i ++) {
		vector<int> nnums;
		for (int i = 0; i < 26; i ++) {
			nnums.emplace_back(nums[i]);
			nnums.emplace_back(nums[i + 26]);
		}
		nums.swap(nnums);

		bool flg = true;
		for (int i = 26; i < 52; i ++) {
			if (nums[i] == 1) {
				flg = false;
			}
		}

		if (flg) return cout << "YES", 0;
	}

	cout << "NO";

	return 0;
}
```
