**提示 1：** 先看火山是怎么活动的。

**提示 2：** 接下来，怎么逃跑最优？

活动会受到火山的限制，所以我们先考虑火山的行动。

这里最重要的就是每个点什么时候会开始被火山影响。这件事等于这个点的数值 / 另一点的数值加到这点的最短路长度。

我们可以理解为从一个虚拟节点出发，先按照 $a_i$ 的边权走到 $i$ ，接下来再走图上的边。因此我们只需在这个图上跑一次最短路，就能得到每个点最早被影响的时间。

接下来我们要从 $1$ 结点出发走到 $n$ ，每个点肯定越早走到越好，因此仍然使用最短路算法。但这里需要多满足一个条件，就是走到一个结点时刻不能晚于这个点被火山影响的时刻。如果在这种情况下能走到 $n$ 结点就满足要求。也就是最短路过程中多个判断。

时间复杂度为 $\mathcal{O}(m+n\log m)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    t = II()
    outs = []
    
    inf = 11 * 10 ** 8
    
    for _ in range(t):
        n, m = MII()
        nums = LII()
        
        path = [[] for _ in range(n)]
        
        for _ in range(m):
            u, v, w = MII()
            u -= 1
            v -= 1
            path[u].append(w * n + v)
            path[v].append(w * n + u)
        
        pq = [nums[i] * n + i for i in range(n)]
        
        while pq:
            d, u = divmod(heappop(pq), n)
            if nums[u] == d:
                for msk in path[u]:
                    nd, v = divmod(msk, n)
                    if nd + d < nums[v]:
                        nums[v] = nd + d
                        heappush(pq, nums[v] * n + v)
        
        dis = [inf] * n
        dis[0] = 0
        
        pq = [0]
        
        while pq:
            d, u = divmod(heappop(pq), n)
            if dis[u] == d:
                for msk in path[u]:
                    nd, v = divmod(msk, n)
                    if nd + d < dis[v] and nd + d <= nums[v]:
                        dis[v] = nd + d
                        heappush(pq, dis[v] * n + v)
        
        outs.append('YES' if dis[n - 1] < inf else 'NO')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t;
	cin >> t;

	int inf = 11e8;

	while (t --) {
		int n, m;
		cin >> n >> m;

		vector<int> nums(n);
		for (auto &v: nums) cin >> v;

		vector<vector<pair<int, int>>> path(n);
		while (m --) {
			int u, v, w;
			cin >> u >> v >> w;
			u --, v --;
			path[u].emplace_back(v, w);
			path[v].emplace_back(u, w);
		}

		priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

		for (int i = 0; i < n; i ++) pq.push({nums[i], i});

		while (!pq.empty()) {
			auto [d, u] = pq.top(); pq.pop();
			if (nums[u] == d) {
				for (auto &[v, w]: path[u]) {
					if (nums[u] + w < nums[v]) {
						nums[v] = nums[u] + w;
						pq.push({nums[v], v});
					}
				}
			}
		}

		vector<int> dis(n, inf);
		dis[0] = 0;

		pq.push({0, 0});

		while (!pq.empty()) {
			auto [d, u] = pq.top(); pq.pop();
			if (dis[u] == d) {
				for (auto &[v, w]: path[u]) {
					if (dis[u] + w < dis[v] && dis[u] + w <= nums[v]) {
						dis[v] = dis[u] + w;
						pq.push({dis[v], v});
					}
				}
			}
		}

		cout << (dis[n - 1] < inf ? "YES\n" : "NO\n");
	}

	return 0;
}
```
