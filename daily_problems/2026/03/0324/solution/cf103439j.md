**提示 1：** 找到最大操作次数。接下来是一些分类讨论的判断。

首先，显然用三次操作，分三段分别变成 A,B,C 可以满足要求。

那两次是否一定可以呢？答案是肯定的。我们只需取最短的前缀，使得其中有字符出现了 $n$ 次，那么我们只需将后缀分为两部分分别变为剩下的两种字符就行了。

$0$ 次的情况又很显然，那么接下来关键就是讨论一次操作能不能完成的情况了。

假设我们将 $[l,r]$ 这个区间变成了 A ，那么外侧的 B,C 数量都是 $n$ 了。所以我们要找到外侧的区间，使得其出现某两个字母的频率都恰好是 $n$ 。

我们可以从小到大枚举 $l$ ，对应的 $r$ 取的越小越好，同时满足外侧的最大频率不超过 $n$ 。这样找到的最小的 $r$ 如果都不满足上面的事情，频率就不可能更大，更大的 $r$ 也就不可能有两个字母都出现 $n$ 次，则该 $l$ 永远不可能找到单次的操作完成任务。

而不同的 $l$ 这里可以单向移动 $r$ ，所以也就顺利解决了。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    n = II()
    s = I()
    
    ca = s.count('A')
    cb = s.count('B')
    cc = s.count('C')
    
    if ca == n and cb == n and cc == n: print(0)
    else:
        l = 0
        r = 0
        
        while l < 3 * n:
            while max(ca, cb, cc) > n and r < 3 * n:
                if s[r] == 'A': ca -= 1
                elif s[r] == 'B': cb -= 1
                else: cc -= 1
                r += 1
            
            if max(ca, cb, cc) > n: break
            
            if ca == n and cb == n:
                print(1)
                print(l + 1, r, 'C')
                exit()
            
            if ca == n and cc == n:
                print(1)
                print(l + 1, r, 'B')
                exit()
            
            if cb == n and cc == n:
                print(1)
                print(l + 1, r, 'A')
                exit()
            
            if s[l] == 'A': ca += 1
            elif s[l] == 'B': cb += 1
            else: cc += 1
            l += 1
        
        ca = 0
        cb = 0
        cc = 0
        
        print(2)
        for i in range(3 * n):
            if s[i] == 'A': ca += 1
            elif s[i] == 'B': cb += 1
            else: cc += 1
            
            if ca == n:
                print(i + 2, 3 * n, 'B')
                print(3 * n - (n - cc) + 1, 3 * n, 'C')
                break
            
            if cb == n:
                print(i + 2, 3 * n, 'A')
                print(3 * n - (n - cc) + 1, 3 * n, 'C')
                break
            
            if cc == n:
                print(i + 2, 3 * n, 'A')
                print(3 * n - (n - cb) + 1, 3 * n, 'B')
                break
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	string s;

	cin >> n >> s;

	int ca = 0, cb = 0, cc = 0;
	for (auto &c: s) {
		if (c == 'A') ca ++;
		else if (c == 'B') cb ++;
		else cc ++;
	}

	if (ca == n && cb == n && cc == n) cout << 0;
	else {
		int l = 0, r = 0;
		while (l < 3 * n) {
			while (max({ca, cb, cc}) > n && r < 3 * n) {
				if (s[r] == 'A') ca --;
				else if (s[r] == 'B') cb --;
				else cc --;
				r ++;
			}

			if (max({ca, cb, cc}) > n) break;

			if (ca == n && cb == n) {
				cout << 1 << '\n';
				cout << l + 1 << ' ' << r << " C\n";
				return 0;
			}

			if (ca == n && cc == n) {
				cout << 1 << '\n';
				cout << l + 1 << ' ' << r << " B\n";
				return 0;
			}

			if (cb == n && cc == n) {
				cout << 1 << '\n';
				cout << l + 1 << ' ' << r << " A\n";
				return 0;
			}

			if (s[l] == 'A') ca ++;
			else if (s[l] == 'B') cb ++;
			else cc ++;

			l ++;
		}

		ca = 0, cb = 0, cc = 0;
		cout << 2 << '\n';

		for (int i = 0; i < 3 * n; i ++) {
			if (s[i] == 'A') ca ++;
			else if (s[i] == 'B') cb ++;
			else cc ++;

			if (ca == n) {
				cout << i + 2 << ' ' << 3 * n << " B\n";
				cout << 3 * n - (n - cc) + 1 << ' ' << 3 * n << " C\n";
				break;
			}

			if (cb == n) {
				cout << i + 2 << ' ' << 3 * n << " A\n";
				cout << 3 * n - (n - cc) + 1 << ' ' << 3 * n << " C\n";
				break;
			}

			if (cc == n) {
				cout << i + 2 << ' ' << 3 * n << " A\n";
				cout << 3 * n - (n - cb) + 1 << ' ' << 3 * n << " B\n";
				break;
			}
		}
	}

	return 0;
}
```
