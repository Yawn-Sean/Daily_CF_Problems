**提示 1：** 每个元素只会操作少量次数。如何找到操作的元素？操作的更新？

首先， $0,1,2$ 可以单独处理，剩余的数越阶乘越大，最后超大了之后就一定超过 $998244353$ 了，再阶乘就变成了 $0$ 。

因此每个元素有用的操作次数（即改变数值的操作次数）都很有限。我们只需找到区间中需要操作的位置就行。

我们可以用有序集合维护所有还能操作的位置，二分找，也可以用线段树维护每个位置的剩余操作次数，线段树二分找到数值大于 $0$ 的第一个位置。

然后修改过程中我们需要维护区间和，因此用另一个树状数组 / 线段树进行维护就好。

同时，这里操作的结果还有一些细节。 $10!,11!,12!$ 这三个数不小，但同时也没到 $998\ 244\ 353$ ，我们需要暴力求一下它们的阶乘进行打表，维护它们操作后会变成啥。别的都只需要 $0\sim 500\ 000$ 的阶乘的预处理就够了。

所以时间复杂度为 $\mathcal{O}(n\log n+q\log n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    M = 5 * 10 ** 5
    mod = 998244353
    
    f = [0] * (M + 1)
    
    f[0] = 1
    for i in range(1, M + 1):
        f[i] = i * f[i - 1] % mod
    
    cnt = [2] * (M + 1)
    cnt[1] = 0
    cnt[2] = 0
    cnt[10] = 3
    cnt[11] = 3
    cnt[12] = 3
    
    for i in range(9, 2, -1):
        cnt[i] = cnt[f[i]] + 1
    
    t = II()
    outs = []
    
    for _ in range(t):
        n, q = MII()
        nums = LII()
        
        tmp = nums[:]
        op_cnt = [cnt[x] for x in nums]
        
        fen = FenwickTreeArray(nums)
        seg = SegTree(fmax, 0, op_cnt)
        
        for _ in range(q):
            o, l, r = MII()
            if o == 1:
                l -= 1
                while True:
                    p = seg.max_right(l, lambda x: x == 0)
                    if p >= r: break
                    
                    op_cnt[p] -= 1
                    if op_cnt[p] == 0:
                        if tmp[p]:
                            fen.add(p, -nums[p])
                            nums[p] = 0
                    
                    elif op_cnt[p] == 1:
                        if tmp[p] == 10:
                            fen.add(p, 818355289)
                            nums[p] = 821984089
                        elif tmp[p] == 11:
                            fen.add(p, 604139442)
                            nums[p] = 644056242
                        elif tmp[p] == 12:
                            fen.add(p, 48654759)
                            nums[p] = 527656359
                        else:
                            fen.add(p, f[nums[p]] - nums[p])
                            nums[p] = f[nums[p]]
                    
                    else:
                        fen.add(p, f[nums[p]] - nums[p])
                        nums[p] = f[nums[p]]
                    
                    seg.set(p, seg.get(p) - 1)
                    l = p + 1
            else:
                outs.append(fen.rsum(l - 1, r - 1) % mod)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int op(int x, int y) {return max(x, y);}

int e() {return 0;}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t, M = 5e5, mod = 998244353;
	cin >> t;

	vector<int> f(M + 1), cnt(M + 1, 2);

	f[0] = 1;
	for (int i = 1; i <= M; i ++)
		f[i] = 1ll * i * f[i - 1] % mod;
	
	cnt[1] = 0;
	cnt[2] = 0;
	cnt[10] = 3;
	cnt[11] = 3;
	cnt[12] = 3;

	for (int i = 9; i > 2; i --) cnt[i] = cnt[f[i]] + 1;

	while (t --) {
		int n, q;
		cin >> n >> q;

		vector<int> nums(n);
		for (auto &v: nums) cin >> v;

		auto tmp = nums;
		vector<int> op_cnt(n);
		for (int i = 0; i < n; i ++) op_cnt[i] = cnt[nums[i]];

		atcoder::fenwick_tree<long long> fen(n);
		atcoder::segtree<int, op, e> seg(op_cnt);

		for (int i = 0; i < n; i ++) fen.add(i, nums[i]);

		while (q --) {
			int o, l, r;
			cin >> o >> l >> r;

			if (o == 1) {
				l --;
				while (true) {
					int p = seg.max_right(l, [&] (int x) {return x == 0;});
					if (p >= r) break;

					op_cnt[p] --;
					if (op_cnt[p] == 0) {
						if (tmp[p]) {
							fen.add(p, -nums[p]);
							nums[p] = 0;
						}
					}
					else if (op_cnt[p] == 1) {
						if (tmp[p] == 10) {
							fen.add(p, 818355289);
							nums[p] = 821984089;
						}
						else if (tmp[p] == 11) {
							fen.add(p, 604139442);
							nums[p] = 644056242;
						}
						else if (tmp[p] == 12) {
							fen.add(p, 48654759);
							nums[p] = 527656359;
						}
						else {
							fen.add(p, f[nums[p]] - nums[p]);
							nums[p] = f[nums[p]];
						}
					}
					else {
						fen.add(p, f[nums[p]] - nums[p]);
						nums[p] = f[nums[p]];
					}
					seg.set(p, seg.get(p) - 1);
					l = p + 1;
				}
			}
			else cout << fen.sum(l - 1, r) % mod << '\n';
		}
	}

	return 0;
}
```
