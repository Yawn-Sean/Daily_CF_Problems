**提示 1：** 先找到必要的条件。可以尝试证明吗？

首先，如果两个数字不互质又产生了逆序对，显然不行——因为迟早有一个操作会让他们顺序反转，而这两个数字又不互质，这个操作就不合法了。

那这个条件充分吗？答案是肯定的，可以基于数学归纳法。最小值一定可以移动到开头，剩余的长度就减去 $1$ 了，就可以用归纳假设了。

而如何判断是否有不互质的数产生了逆序对呢？我们只需对每个质数存是当前这个质数的倍数的最大数值，这样顺序遍历的过程中，遍历到 $x$ ，如果 $x$ 的某个质因子存过比 $x$ 更大的数值，那么就产生了矛盾。

时间复杂度为 $\mathcal{O}(M\log\log M + n\log M)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    M = 10 ** 6 + 5
    pr = list(range(M))
    
    for i in range(2, M):
        if pr[i] == i:
            for j in range(i, M, i):
                pr[j] = i
    
    def prime_factor(x):
        ans = []
        while x > 1:
            p = pr[x]
            ans.append(p)
            while x % p == 0:
                x //= p
        return ans
    
    t = II()
    outs = []
    
    vis = [0] * M
    
    for _ in range(t):
        n = II()
        nums = LII()
        
        flg = True
        
        for x in nums:
            for p in prime_factor(x):
                if vis[p] > x:
                    flg = False
                vis[p] = x
        
        outs.append('SI' if flg else 'NO')
        
        for x in nums:
            for p in prime_factor(x):
                vis[p] = 0
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e6 + 5;
	vector<int> pr(M);
	iota(pr.begin(), pr.end(), 0);

	for (int i = 2; i < M; i ++) {
		if (pr[i] == i) {
			for (int j = i; j < M; j += i) {
				pr[j] = i;
			}
		}
	}

	auto prime_factors = [&] (int x) -> vector<int> {
		vector<int> ans;
		while (x > 1) {
			int p = pr[x];
			ans.emplace_back(p);
			while (x % p == 0) x /= p;
		}
		return ans;
	};

	int t;
	cin >> t;

	vector<int> vis(M, 0);

	while (t --) {
		int n;
		cin >> n;

		vector<int> nums(n);
		for (auto &x: nums) cin >> x;

		bool flg = true;
		for (auto &x: nums) {
			for (auto &p: prime_factors(x)) {
				if (vis[p] > x) flg = false;
				vis[p] = x;
			}
		}

		cout << (flg ? "SI\n" : "NO\n");

		for (auto &x: nums) {
			for (auto &p: prime_factors(x)) {
				vis[p] = 0;
			}
		}
	}

	return 0;
}
```
