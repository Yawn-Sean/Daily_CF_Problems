**提示 1：** 相当于在图中找路径。什么图？怎么快速构造出一条路径？

把一个电影看成一个结点，一个人也看成一个结点。如果出演了的话就连一条边。这样问题就变成了找两个演员之间的路径。

如果两个演员不在一个连通块内，显然不存在路径。

否则如何找到路径呢？我们对一个连通块，我们可以取一个最小生成树，这样，树上路径就是很容易得到的了。

这条路径的长度怎么保证不过长呢？我们可以 BFS ，因为任意两个结点之间的距离最多经过 $n$ 个电影，因此距离是不超过 $\mathcal{O}(n)$ 的，也就保证了求出来的路径长度不长。（DFS 其实也是可以的，因为也最多到 $\mathcal{O}(n)$ 层，因为每两层总有一层是电影，而电影总共只有 $n$ 个。）

时间复杂度为 $\mathcal{O}(n+m+qn)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    n, m = MII()
    uf = UnionFind(n + m)
    
    path = [[] for _ in range(n + m)]
    
    for i in range(n):
        _, *idxs = MII()
        
        for idx in idxs:
            idx -= 1
            if uf.merge(idx, m + i):
                path[idx].append(m + i)
                path[m + i].append(idx)
    
    parent = [-1] * (n + m)
    depth = [0] * (n + m)
    
    for i in range(n + m):
        if parent[i] == -1:
            que = [i]
            for u in que:
                for v in path[u]:
                    if parent[u] != v:
                        parent[v] = u
                        depth[v] = depth[u] + 1
                        que.append(v)
    
    q = II()
    outs = []
    
    for _ in range(q):
        u, v = GMI()
        
        if uf.find(u) != uf.find(v): outs.append('-1')
        else:
            path_u = [u]
            path_v = [v]
            
            while path_u[-1] != path_v[-1]:
                if depth[path_u[-1]] > depth[path_v[-1]]:
                    path_u.append(parent[path_u[-1]])
                else:
                    path_v.append(parent[path_v[-1]])
            
            path_v.reverse()
            path_u.extend(path_v[1:])
            
            outs.append(str(len(path_u) // 2 + 1))
            outs.append(' '.join(str(x + 1) if x < m else str(x - m + 1) for x in path_u))
    
    print('\n'.join(outs))

```

C++ 做法如下——

```cpp []

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m;
	cin >> n >> m;

	atcoder::dsu uf(n + m);
	vector<vector<int>> path(n + m);

	for (int i = 0; i < n; i ++) {
		int k;
		cin >> k;

		while (k --) {
			int u;
			cin >> u;
			u --;

			if (uf.merge(u, m + i)) {
				path[u].emplace_back(m + i);
				path[m + i].emplace_back(u);
			}
		}
	}

	vector<int> parent(n + m, -1), depth(n + m, 0);

	auto dfs = [&] (auto &self, int u, int p) -> void {
		for (auto &v: path[u]) {
			if (v != p) {
				parent[v] = u;
				depth[v] = depth[u] + 1;
				self(self, v, u);
			}
		}
	};

	for (int i = 0; i < n + m; i ++) {
		if (parent[i] == -1) {
			dfs(dfs, i, -1);
		}
	}

	int q;
	cin >> q;

	while (q --) {
		int u, v;
		cin >> u >> v;
		u --, v --;

		if (uf.leader(u) != uf.leader(v)) cout << "-1\n";
		else {
			vector<int> path_u = {u}, path_v = {v};

			while (path_u.back() != path_v.back()) {
				if (depth[path_u.back()] > depth[path_v.back()])
					path_u.emplace_back(parent[path_u.back()]);
				else path_v.emplace_back(parent[path_v.back()]);
			}

			path_u.insert(path_u.end(), path_v.rbegin() + 1, path_v.rend());
			cout << path_u.size() / 2 + 1 << '\n';

			for (auto &x: path_u) cout << (x < m ? x + 1 : x - m + 1) << ' ';
			cout << '\n';
		}
	}

	return 0;
}
```
