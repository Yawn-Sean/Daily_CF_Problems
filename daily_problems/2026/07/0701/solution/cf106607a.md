**提示 1：** 静态问题我们做过了，咋做？

**提示 2：** 从第 $i$ 个答案推到第 $i+1$ 个答案。

首先，如果只需求最终的答案，怎么做呢？

先考虑将对应的数字连边，则每条边需要钦定一个互不相同的端点。

每个连通块的方案相乘即可。

如果边数大于点数，则至少有一点分配给两边（抽屉原理），矛盾。

如果边数等于点数，则恰好有一个环，且环上的点只能分给环上的边，不然的话外侧的边数就大于点数了。而环的分配方式有两种。

如果边数等于点数减一，则整个形成一棵树，钦定树上任意一点不被选都对应了一种方案，因此答案是点的个数。

而在连边的过程中只需维护当前两个点连边造成的影响，即可通过前一轮的答案计算下一轮的答案：

- 将两棵树相连：除以它们原本的大小，乘以后来的大小。

- 有环的连通块和树相连：除以树的原先的大小。

- 将两个有环的连通块相连：答案变为 $0$ 。

- 在一个连通块内连边：

	- 如果本身是树，则除以树的大小，乘以 $2$ 。

	- 如果本身有环，则答案变为 $0$ 。

这里预处理逆元就可以快速完成这件事。

是否有环可以在并查集的维护过程中，在对应连通块的结点打标记实现。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    t = II()
    outs = []
    
    mod = 998244353
    
    M = 3 * 10 ** 5 + 5
    invs = [0] * M
    
    invs[1] = 1
    
    for i in range(2, M):
        invs[i] = (mod - mod // i) * invs[mod % i] % mod
    
    for _ in range(t):
        n = II()
        us = []
        vs = []
        
        for _ in range(n):
            u, v = MII()
            us.append(u)
            vs.append(v)
        
        nodes = sorted(us + vs)
        
        for i in range(n):
            us[i] = bisect.bisect_left(nodes, us[i])
            vs[i] = bisect.bisect_left(nodes, vs[i])
        
        res = []
        ans = 1
        
        tag = [0] * (n * 2)
        uf = UnionFind(n * 2)
        
        for i in range(n):
            u = uf.find(us[i])
            v = uf.find(vs[i])
            szu = uf.getSize(u)
            szv = uf.getSize(v)
            
            uf.merge(u, v)
            p = uf.find(u)
            
            if u != v:
                if tag[u] and tag[v]:
                    ans = 0
                elif tag[u]:
                    ans = ans * invs[szv] % mod
                    tag[p] = 1
                elif tag[v]:
                    ans = ans * invs[szu] % mod
                    tag[p] = 1
                else:
                    ans = ans * invs[szu] % mod * invs[szv] % mod * (szu + szv) % mod
            else:
                if tag[u]:
                    ans = 0
                else:
                    ans = ans * invs[szu] % mod * 2 % mod
                    tag[u] = 1
            
            res.append(ans)
        
        outs.append(' '.join(map(str, res)))
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 3e5 + 5, mod = 998244353;
	vector<int> invs(M);
	
	invs[1] = 1;
	for (int i = 2; i < M; i ++)
		invs[i] = 1ll * (mod - mod / i) * invs[mod % i] % mod;

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;

		vector<pair<int, int>> edges(n);
		vector<int> nodes;

		for (auto &[u, v]: edges) {
			cin >> u >> v;
			nodes.emplace_back(u);
			nodes.emplace_back(v);
		}

		sort(nodes.begin(), nodes.end());

		atcoder::dsu uf(2 * n);
		vector<int> tag(2 * n, 0);

		int ans = 1;

		for (int i = 0; i < n; i ++) {
			int u = lower_bound(nodes.begin(), nodes.end(), edges[i].first) - nodes.begin();
			int v = lower_bound(nodes.begin(), nodes.end(), edges[i].second) - nodes.begin();

			u = uf.leader(u), v = uf.leader(v);
			int szu = uf.size(u), szv = uf.size(v);

			uf.merge(u, v);
			int p = uf.leader(u);

			if (u != v) {
				if (tag[u] && tag[v]) ans = 0;
				else if (tag[u]) {
					ans = 1ll * ans * invs[szv] % mod;
					tag[p] = 1;
				}
				else if (tag[v]) {
					ans = 1ll * ans * invs[szu] % mod;
					tag[p] = 1;
				}
				else {
					ans = 1ll * ans * invs[szu] % mod * invs[szv] % mod * (szu + szv) % mod;
				}
			}
			else {
				if (tag[u]) ans = 0;
				else {
					ans = 2ll * ans * invs[szu] % mod;
					tag[u] = 1;
				}
			}

			cout << ans << " \n"[i == n - 1];
		}
	}

	return 0;
}
```
