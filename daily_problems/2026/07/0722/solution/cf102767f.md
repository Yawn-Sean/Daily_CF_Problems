**提示 1：** 对于每个右端点，枚举每个答案看能否达成。

一个区间的乘积关于 $x$ 的取模结果，可以用前缀乘积表示。右端点对应位置的前缀积乘以左端点对应位置的前缀积的逆元再关于 $x$ 取模即可。

所以我们枚举右端点的时候，我们只在乎左端点的各种取模结果的最近一次出现。而这可以用数组来存，对应的区间乘积的结果可以用前面逆元的方式得到，而每次更新也很方便。

时间复杂度为 $\mathcal{O}(nx)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    t = II()
    outs = []
    
    for _ in range(t):
        n, x = MII()
        nums = LII()
        
        mod_inv = [0] * x
        mod_inv[1] = 1
        
        for i in range(2, x):
            mod_inv[i] = mod_inv[x % i] * (x - x // i) % x
        
        ans = 0
        cur_len = 0
        
        vis = [-2] * x
        vis[1] = -1
        
        cur = 1
        
        for i in range(n):
            cur = cur * nums[i] % x
            
            for j in range(x):
                if vis[j] != -2 and (cur * mod_inv[j] % x > ans or (cur * mod_inv[j] % x == ans and i - vis[j] < cur_len)):
                    ans = cur * mod_inv[j] % x
                    cur_len = i - vis[j]
            
            vis[cur] = i
        
        outs.append(f'{ans} {cur_len}')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t;
	cin >> t;

	while (t --) {
		int n, x;
		cin >> n >> x;

		vector<int> nums(n);
		for (auto &v: nums) cin >> v;

		vector<int> mod_inv(x);
		mod_inv[1] = 1;

		for (int i = 2; i < x; i ++)
			mod_inv[i] = mod_inv[x % i] * (x - x / i) % x;
		
		int ans = 0, cur_len = 0;
		vector<int> vis(x, -2);
		vis[1] = -1;

		int cur = 1;

		for (int i = 0; i < n; i ++) {
			cur = cur * nums[i] % x;

			for (int j = 0; j < x; j ++) {
				if (vis[j] != -2 && (cur * mod_inv[j] % x > ans || (cur * mod_inv[j] % x == ans && i - vis[j] < cur_len))) {
					ans = cur * mod_inv[j] % x;
					cur_len = i - vis[j];
				}
			}

			vis[cur] = i;
		}

		cout << ans << ' ' << cur_len << '\n';
	}

	return 0;
}
```
