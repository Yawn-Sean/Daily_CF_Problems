**提示 1：** 考虑一对元素的贡献。

时间复杂度为 $\mathcal{O}(n+m)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    t = II()
    outs = []
    mod = 10 ** 9 + 7
    
    for _ in range(t):
        n, m = MII()
        
        tmp1 = [0] * (m + 1)
        tmp2 = [0] * (m + 1)
        
        for _ in range(n):
            l, r = MII()
            tmp1[l] += m + 1 - r
            tmp2[r] += l
        
        ans = 0
        total = 0
        
        for i in range(m, 0, -1):
            ans = (ans + tmp2[i] * total) % mod
            total = (total + tmp1[i]) % mod
        
        outs.append(ans)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e5 + 5;
	vector<int> cnt(M, 0);

	for (int i = 1; i < M; i ++) {
		for (int j = i * 2; j < M; j += i) {
			cnt[j] ++;
		}
	}

	for (int i = 1; i < M; i ++) cnt[i] += cnt[i - 1];

	int mod = 1e9 + 7, rev2 = (mod + 1) / 2;
	vector<int> f(M, 1);
	for (int i = 1; i < M; i ++) f[i] = 1ll * i * f[i - 1] % mod;

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;
		cout << 1ll * cnt[n] * rev2 % mod * f[n] % mod << '\n';
	}

	return 0;
}
```
