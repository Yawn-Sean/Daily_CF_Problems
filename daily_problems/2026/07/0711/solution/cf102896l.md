**提示 1：** 答案其实可以由 “分裂继续查” 的结点的个数决定。

**提示 2：** 如何找到对应结点。

整体被查的点也形成一棵二叉树。所以我们只需考虑这棵二叉树种非叶子节点的个数，即 “分裂继续查” 的结点的个数。

于是我们需要找到不包含于 $[l,r]$ 区间的子树个数。

每一个深度被查的其实最多只有两个结点，所以其实我们只需知道左侧的那个结点最深到哪，右侧的结点最深到哪里，则所有根节点到两个结点之一的路径上的点都加入计数即可。

考虑所有结点对应的数值，则左侧的那个结点应该至少包含一个小于 $l$ 的树，一个大于等于 $l$ 的数。所以找到这两个数对应的结点，求 LCA ，这个点就是左侧结点。

右侧结点是类似的。

而总标记的结点个数可以用两个节点的深度和减去这两个结点 LCA 的深度得到。

注意答案是 $1$ 的情况的特判。

时间复杂度为 $\mathcal{O}(n\log n)$ 。可以有线性做法。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    vs = []
    
    parent = [-1] * n
    
    for i in range(n):
        l, r, v = MII()
        l -= 1
        r -= 1
        
        if l >= 0: parent[l] = i
        if r >= 0: parent[r] = i
        
        vs.append(v)
    
    depth = [0] * n
    
    for i in range(1, n):
        depth[i] = depth[parent[i]] + 1
    
    st_range = sorted(range(n), key=lambda x: vs[x])
    sorted_vs = sorted(vs)
    
    nth_parent = [[-1] * n for _ in range(20)]
    
    nth_parent[0] = parent
    
    for i in range(19):
        for j in range(n):
            if nth_parent[i][j] != -1:
                nth_parent[i + 1][j] = nth_parent[i][nth_parent[i][j]]
    
    def lca(u, v):
        if depth[u] > depth[v]: u, v = v, u
        d = depth[v] - depth[u]
        
        while d:
            x = d & -d
            bit = x.bit_length() - 1
            v = nth_parent[bit][v]
            d -= x
        
        if u == v: return u
        
        for i in range(19, -1, -1):
            if nth_parent[i][u] != nth_parent[i][v]:
                u = nth_parent[i][u]
                v = nth_parent[i][v]
        
        return parent[u]
    
    q = II()
    outs = []
    
    for _ in range(q):
        l, r = MII()
        
        if l > sorted_vs[-1]: outs.append(1)
        elif r < sorted_vs[0]: outs.append(1)
        else:
            pl = bisect.bisect_left(sorted_vs, l)
            pr = bisect.bisect_right(sorted_vs, r)
            
            if pl == 0: vl = 0
            else: vl = lca(st_range[pl], st_range[pl - 1])
            
            if pr == n: vr = 0
            else: vr = lca(st_range[pr], st_range[pr - 1])
            
            p = lca(vl, vr)
            if pl == 0 and pr == n: outs.append(1)
            else: outs.append(2 * (depth[vl] + depth[vr] - depth[p]) + 3)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	vector<int> parent(n, -1), vs(n);

	for (int i = 0; i < n; i ++) {
		int l, r, v;
		cin >> l >> r >> v;

		if (l) parent[l - 1] = i;
		if (r) parent[r - 1] = i;

		vs[i] = v;
	}

	vector<int> depth(n, 0);

	for (int i = 1; i < n; i ++) depth[i] = depth[parent[i]] + 1;

	vector<int> st_range(n), sorted_vs = vs;
	iota(st_range.begin(), st_range.end(), 0);
	sort(st_range.begin(), st_range.end(), [&] (int i, int j) {
		return vs[i] < vs[j];
	});
	sort(sorted_vs.begin(), sorted_vs.end());

	vector<vector<int>> nth_parent(20, vector<int>(n, -1));

	nth_parent[0] = parent;

	for (int i = 0; i < 19; i ++) {
		for (int j = 0; j < n; j ++) {
			if (nth_parent[i][j] != -1) {
				nth_parent[i + 1][j] = nth_parent[i][nth_parent[i][j]];
			}
		}
	}

	auto lca = [&] (int u, int v) -> int {
		if (depth[u] > depth[v]) swap(u, v);

		int d = depth[v] - depth[u];

		while (d) {
			v = nth_parent[__builtin_ctz(d)][v];
			d -= d & -d;
		}

		if (u == v) return u;

		for (int i = 19; i >= 0; i --) {
			if (nth_parent[i][u] != nth_parent[i][v]) {
				u = nth_parent[i][u];
				v = nth_parent[i][v];
			}
		}

		return parent[u];
	};

	int q;
	cin >> q;

	while (q --) {
		int l, r;
		cin >> l >> r;

		if (l > sorted_vs.back()) cout << 1 << '\n';
		else if (r < sorted_vs[0]) cout << 1 << '\n';
		else {
			int pl = lower_bound(sorted_vs.begin(), sorted_vs.end(), l) - sorted_vs.begin();
			int pr = upper_bound(sorted_vs.begin(), sorted_vs.end(), r) - sorted_vs.begin();

			int vl = pl == 0 ? 0 : lca(st_range[pl], st_range[pl - 1]);
			int vr = pr == n ? 0 : lca(st_range[pr], st_range[pr - 1]);

			int p = lca(vl, vr);

			if (pl == 0 && pr == n) cout << 1 << '\n';
			else cout << 2 * (depth[vl] + depth[vr] - depth[p]) + 3 << '\n';
		}
	}

	return 0;
}
```
