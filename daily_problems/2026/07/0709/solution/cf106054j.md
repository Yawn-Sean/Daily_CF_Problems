**提示 1：** $t\geq 3$ 的时候可以任意挪动。剩余情况呢？

首先，两组东西的元素构成必须完全一致，因为挪动不改变这件事。接下来考虑满足这件事的情况。

而 $t\geq 3$ 时，因为用数学归纳法，我们可以把前 $k$ 小的元素重新排成一堆，因此两组东西都可以转化为排好序的一堆。而操作可逆，所以两个状态总是可以转化。

$t=1$ ，只有一堆，已经自然成立。

$t=2$ ，此时只可能是尾部一些元素的交换。我们用第一个序列拼接上第二个序列的反转，则构成的序列需要完全一致，因为这个序列在过程中也保持不变。

如果满足不变，则可以转化；否则不能。

时间复杂度为 $\mathcal{O}(n)$ ，如果你线性判断元素构成是否相同。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    t = II()
    
    v1 = [[] for _ in range(t)]
    w1 = []
    
    for i in range(t):
        k, *v1[i] = MII()
        w1.extend(v1[i])
    
    v2 = [[] for _ in range(t)]
    w2 = []
    for i in range(t):
        k, *v2[i] = MII()
        w2.extend(v2[i])
    
    w1.sort()
    w2.sort()
    
    if w1 != w2: print('N')
    elif t == 1 or t >= 3: print('S')
    else:
        while v1[1]:
            v1[0].append(v1[1].pop())
        
        while v2[1]:
            v2[0].append(v2[1].pop())
        
        if v1[0] == v2[0]: print('S')
        else: print('N')
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

    int t;
    cin >> t;

    vector<vector<int>> v1(t);
    vector<int> w1;

    for (int i = 0; i < t; i ++) {
        int k;
        cin >> k;
        while (k --) {
            int x;
            cin >> x;
            v1[i].emplace_back(x);
            w1.emplace_back(x);
        }
    }

    vector<vector<int>> v2(t);
    vector<int> w2;

    for (int i = 0; i < t; i ++) {
        int k;
        cin >> k;
        while (k --) {
            int x;
            cin >> x;
            v2[i].emplace_back(x);
            w2.emplace_back(x);
        }
    }

    auto equals = [&] (vector<int> &v1, vector<int> &v2) -> bool {
        int k1 = v1.size(), k2 = v2.size();
        if (k1 != k2) return false;

        for (int i = 0; i < k1; i ++) {
            if (v1[i] != v2[i]) {
                return false;
            }
        }

        return true;
    };

    sort(w1.begin(), w1.end());
    sort(w2.begin(), w2.end());

    if (!equals(w1, w2)) cout << 'N';
    else if (t == 1 || t >= 3) cout << 'S';
    else {
        while (!v1[1].empty()) {
            auto v = v1[1].back();
            v1[1].pop_back();
            v1[0].emplace_back(v);
        }
        while (!v2[1].empty()) {
            auto v = v2[1].back();
            v2[1].pop_back();
            v2[0].emplace_back(v);
        }

        if (equals(v1[0], v2[0])) cout << 'S';
        else cout << 'N';
    }

	return 0;
}
```
