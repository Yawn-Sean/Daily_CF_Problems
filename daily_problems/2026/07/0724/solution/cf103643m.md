**提示 1：** 考虑相较于不删点而言答案怎么变。

首先，我们先统计初始情况下，度数不超过 $1$ 的点的个数。

接下来考虑删除两个点带来的影响。

可能会删掉已经被统计进答案的点，我们去掉它们。

其它的，只对度数不超过 $3$ 的点有影响。这样的点实际个数是 $\mathcal{O}(\frac{m}{n})$ 的，因为总共只删了 $m$ 条边，而这样的点需要删掉对应的 $\mathcal{O}(n)$ 条边。

对于这些点，可以 $\mathcal{O}(n)$ 找到其邻接的点，因此总复杂度是 $\mathcal{O}(\frac{m}{n}\times n)=\mathcal{O}(m)$ 。

- 如果度数是 $2$ ，那么删去任何一个端点 / 两个端点，都会使得答案 $+1$ 。

- 如果度数是 $3$ ，需要删去三点之二才能使答案 $+1$ 。

注意上述两个调整即可。

时间复杂度为 $\mathcal{O}(n+m+q)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, m, q = MII()
    
    degs = [n - 1] * n
    path = [[] for _ in range(n)]
    
    for _ in range(m):
        u, v = GMI()
        
        degs[u] -= 1
        degs[v] -= 1
        
        path[u].append(v)
        path[v].append(u)
    
    vis = [0] * n
    
    base = 0
    diff = [0] * n
    
    for i in range(n):
        if degs[i] <= 1:
            base += 1
            diff[i] -= 1
    
    pairs = Counter()
    
    for i in range(n):
        if 2 <= degs[i] <= 3:
            vis[i] = 1
            for j in path[i]:
                vis[j] = 1
            
            tmp = []
            for j in range(n):
                if not vis[j]:
                    tmp.append(j)
            
            if degs[i] == 2:
                u, v = tmp
                diff[u] += 1
                diff[v] += 1
                pairs[(u, v)] -= 1
                
                if i < u: pairs[(i, u)] -= 1
                else: pairs[(u, i)] -= 1
                
                if i < v: pairs[(i, v)] -= 1
                else: pairs[(v, i)] -= 1
            else:
                u, v, w = tmp
                pairs[(u, v)] += 1
                pairs[(v, w)] += 1
                pairs[(u, w)] += 1
            
            vis[i] = 0
            for j in path[i]:
                vis[j] = 0
    
    outs = []
    
    for _ in range(q):
        u, v = GMI()
        if u > v: u, v = v, u
        
        ans = base + diff[u] + diff[v]
        if (u, v) in pairs: ans += pairs[(u, v)]
        
        outs.append(ans)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m, q;
	cin >> n >> m >> q;

	vector<int> degs(n, n - 1);
	vector<vector<int>> path(n);

	while (m --) {
		int u, v;
		cin >> u >> v;
		u --, v --;

		degs[u] --, degs[v] --;
		path[u].emplace_back(v);
		path[v].emplace_back(u);
	}

	vector<int> vis(n, 0);

	int base = 0;
	vector<int> diff(n, 0);

	for (int i = 0; i < n; i ++) {
		if (degs[i] <= 1) {
			base ++;
			diff[i] --;
		}
	}

	map<pair<int, int>, int> pairs;

	for (int i = 0; i < n; i ++) {
		if (degs[i] >= 2 && degs[i] <= 3) {
			vis[i] = 1;
			for (auto &j: path[i]) vis[j] = 1;

			vector<int> tmp;
			for (int j = 0; j < n; j ++) {
				if (!vis[j]) {
					tmp.emplace_back(j);
				}
			}

			if (degs[i] == 2) {
				int u = tmp[0], v = tmp[1];

				diff[u] ++;
				diff[v] ++;
				pairs[{u, v}] --;

				if (i < u) pairs[{i, u}] --;
				else pairs[{u, i}] --;

				if (i < v) pairs[{i, v}] --;
				else pairs[{v, i}] --;
			}
			else {
				int u = tmp[0], v = tmp[1], w = tmp[2];
				pairs[{u, v}] ++;
				pairs[{u, w}] ++;
				pairs[{v, w}] ++;
			}

			vis[i] = 0;
			for (auto &j: path[i]) vis[j] = 0;
		}
	}

	while (q --) {
		int u, v;
		cin >> u >> v;
		u --, v --;

		if (u > v) swap(u, v);

		int ans = base + diff[u] + diff[v];
		if (pairs.find({u, v}) != pairs.end()) ans += pairs[{u, v}];

		cout << ans << '\n';
	}

	return 0;
}
```
