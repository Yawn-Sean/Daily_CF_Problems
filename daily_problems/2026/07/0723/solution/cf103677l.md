**提示 1：** 考虑每个子集出现的概率，设为 $dp[msk]$ 。

看提示。

对于每个子集，我们枚举操作的两个元素，将对应的概率转移到删去某个元素的新的状态即可。转移概率乘以 $grid[i][j]$ 再除以总二元组数量（任何一个二元组选取都是等概率的）。

时间复杂度为 $\mathcal{O}(n^22^n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    grid = [LFI() for _ in range(n)]
    
    dp = [0] * (1 << n)
    dp[-1] = 1
    
    for i in range((1 << n) - 1, 0, -1):
        v = i.bit_count()
        total = v * (v - 1) // 2
        
        for x in range(n):
            for y in range(x):
                if i >> x & 1 and i >> y & 1:
                    ni = i ^ (1 << x)
                    dp[ni] += dp[i] * grid[y][x] / total
                    ni = i ^ (1 << y)
                    dp[ni] += dp[i] * grid[x][y] / total
    
    print(*(dp[1 << i] for i in range(n)), sep='\n')
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	vector<vector<long double>> grid(n, vector<long double>(n));
	for (auto &v: grid) for (auto &x: v) cin >> x;

	vector<long double> dp(1 << n, 0);
	dp.back() = 1;

	for (int i = (1 << n) - 1; i > 0; i --) {
		int c = __popcount(i), total = c * (c - 1) / 2;

		for (int j = 0; j < n; j ++) {
			for (int k = 0; k < j; k ++) {
				if ((i >> j & 1) && (i >> k & 1)) {
					dp[i ^ (1 << j)] += dp[i] * grid[k][j] / total;
					dp[i ^ (1 << k)] += dp[i] * grid[j][k] / total;
				}
			}
		}
	}

	cout << fixed << setprecision(15);
	for (int i = 0; i < n; i ++) cout << dp[1 << i] << '\n';

	return 0;
}
```
