**提示 1：** 如果每个元素都可变，似乎很好构造。

**提示 2：** 如果有元素不可变，那么最后的最大公约数似乎可能性很少。

如果每个数都可变，则每个数都可以变成偶数，也就解决问题了。

如果存在一个数是不可变的，那么最后的最大公约数一定是这个数的因子。进一步的，如果不是 $1$ 的话，一定是这个数的某个质因子的倍数。

所以我们将这个数质因数分解，看其能否作为最终的最大公约数的因子。即判断每个区间中是否存在一个 $p$ 的倍数，这件事可以找最大的可能的 $p$ 的倍数 $\lfloor r_i/p\rfloor\times p$ ，判断它是否大于等于 $l_i$ 。

时间复杂度为 $\mathcal{O}(n\log M+M)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    M = 10 ** 6
    pr = list(range(M + 1))
    
    for i in range(2, M + 1):
        if pr[i] == i:
            for j in range(i, M + 1, i):
                pr[j] = i
    
    t = II()
    outs = []
    
    for _ in range(t):
        n = II()
        ls = LII()
        rs = LII()
        
        to_check = -1
        for i in range(n):
            if ls[i] == rs[i]:
                to_check = ls[i]
        
        if to_check == -1:
            outs.append('YES')
            outs.append(' '.join(str(rs[i] // 2 * 2) for i in range(n)))
        
        else:
            prs = set()
            
            while to_check > 1:
                prs.add(pr[to_check])
                to_check //= pr[to_check]
            
            for p in prs:
                flg = True
                for i in range(n):
                    if rs[i] // p * p < ls[i]:
                        flg = False
                
                if flg:
                    outs.append('YES')
                    outs.append(' '.join(str(rs[i] // p * p) for i in range(n)))
                    break
            else:
                outs.append('NO')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e6;
	vector<int> pr(M + 1);
	iota(pr.begin(), pr.end(), 0);

	for (int i = 2; i <= M; i ++) {
		if (pr[i] == i) {
			for (int j = i; j <= M; j += i) {
				pr[j] = i;
			}
		}
	}

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;

		vector<int> ls(n), rs(n);
		for (auto &x: ls) cin >> x;
		for (auto &x: rs) cin >> x;

		int to_check = -1;
		for (int i = 0; i < n; i ++) {
			if (ls[i] == rs[i]) {
				to_check = ls[i];
			}
		}

		if (to_check == -1) {
			cout << "YES\n";
			for (int i = 0; i < n; i ++)
				cout << rs[i] / 2 * 2 << ' ';
			cout << '\n';
		}
		else {
			set<int> prs;
			while (to_check > 1) {
				prs.insert(pr[to_check]);
				to_check /= pr[to_check];
			}

			bool found = false;
			for (auto &p: prs) {
				bool flg = true;
				for (int i = 0; i < n; i ++) {
					if (rs[i] / p * p < ls[i]) {
						flg = false;
					}
				}
				if (flg) {
					found = true;
					cout << "YES\n";
					for (int i = 0; i < n; i ++)
						cout << rs[i] / p * p << ' ';
					cout << '\n';
					break;
				}
			}
			if (!found) cout << "NO\n";
		}
	}

	return 0;
}
```
