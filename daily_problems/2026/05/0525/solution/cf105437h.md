**提示 1：** 前缀和后缀是不变的。所以大量数值不变，考虑中间如何调整。

先看提示。

我们先看完全不操作的情况下的答案。这个直接维护到当前位置为止的最后一个字符及其出现次数就够了。

在交换的时候，我们会断开两段，再重新拼接两段。所以我们减去断开两段带来的成本，加上拼接两段带来的收益即可。

这个成本和收益仅仅取决于是否等于拼接 / 拆开的元素是否相同，以及这一段的长度。所以我们可以预处理顺序到某个位置的连续相同元素个数，以及逆序回来到某个位置的连续相同元素个数。

如果把 $x+y$ 拆成 $x,y$ 两段，那么成本是 $(x+y)^2-x^2-y^2=2xy$ ；拼接也是类似的。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    s = [ord(c) - ord('a') for c in I()]
    
    pref_ch = [0] * (n + 1)
    pref_cnt = [0] * (n + 1)
    
    ch = 0
    cnt = 0
    cur = 0
    
    for i in range(n):
        if s[i] != ch:
            ch = s[i]
            cnt = 0
        
        cur += 2 * cnt + 1
        cnt += 1
        
        pref_ch[i + 1] = ch
        pref_cnt[i + 1] = cnt
    
    suff_ch = [0] * (n + 1)
    suff_cnt = [0] * (n + 1)
    
    ch = 0
    cnt = 0
    
    for i in range(n - 1, -1, -1):
        if s[i] != ch:
            ch = s[i]
            cnt = 0
        
        cnt += 1
        
        suff_ch[i] = ch
        suff_cnt[i] = cnt
    
    ans = 0
    
    for i in range(1, n):
        if s[i] != s[i - 1]:
            val = cur
    
            val -= (pref_cnt[i - 1] if s[i - 1] == pref_ch[i - 1] else 0) * 2
            val -= (suff_cnt[i + 1] if s[i] == suff_ch[i + 1] else 0) * 2
            
            val += (pref_cnt[i - 1] if s[i] == pref_ch[i - 1] else 0) * 2
            val += (suff_cnt[i + 1] if s[i - 1] == suff_ch[i + 1] else 0) * 2
            
            ans = fmax(ans, val)
    
    print(ans)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	string s;

	cin >> n >> s;

	vector<int> pref_ch(n + 1, 0);
	vector<int> pref_cnt(n + 1, 0);

	vector<int> suff_ch(n + 1, 0);
	vector<int> suff_cnt(n + 1, 0);

	int ch = 0, cnt = 0;
	long long cur = 0;

	for (int i = 0; i < n; i ++) {
		if (s[i] - 'a' != ch) ch = s[i] - 'a', cnt = 0;

		cur += 2 * cnt + 1;
		cnt ++;

		pref_ch[i + 1] = ch;
		pref_cnt[i + 1] = cnt;
	}

	ch = 0, cnt = 0;

	for (int i = n - 1; i >= 0; i --) {
		if (s[i] - 'a' != ch) ch = s[i] - 'a', cnt = 0;

		cnt ++;

		suff_ch[i] = ch;
		suff_cnt[i] = cnt;
	}

	long long ans = 0;

	for (int i = 1; i < n; i ++) {
		if (s[i] != s[i - 1]) {
			long long val = cur;

			val -= 2 * (s[i - 1] - 'a' == pref_ch[i - 1] ? pref_cnt[i - 1] : 0);
			val -= 2 * (s[i] - 'a' == suff_ch[i + 1] ? suff_cnt[i + 1] : 0);

			val += 2 * ((s[i] - 'a' == pref_ch[i - 1]) ? pref_cnt[i - 1] : 0);
			val += 2 * (s[i - 1] - 'a' == suff_ch[i + 1] ? suff_cnt[i + 1] : 0);

			ans = max(ans, val);
		}
	}

	cout << ans;

	return 0;
}
```
