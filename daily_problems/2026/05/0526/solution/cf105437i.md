**提示 1：** 到每次检查的时候有什么状态？

到每次检查时，前面分配了 $x$ 次点数，所以我们只要知道有 $y$ 的力量值，则智力值一定是 $x-y$ 。所以维护的时候只需要维护其一即可。考虑到第 $i$ 个位置，力量值为 $y$ 时的最大通过检查次数为 $dp[i][y]$ 。

每次新来一个分配点数时，只需枚举这个点数是分配给哪一个进行暴力状态转移即可。

而如果是检查点，则会给一个区间进行答案的 $+1$ 操作，这很容易差分维护，等到下次新增分配点数 / 最后获取答案时再更新即可。

时间复杂度为 $\mathcal{O}(n+m^2)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, m = MII()
    nums = LII()
    
    cur = 0
    dp = [0]
    diff = [0] * 2
    
    def update():
        global cur
        cur += 1
        
        for i in range(cur):
            diff[i + 1] += diff[i]
        
        for i in range(cur):
            dp[i] += diff[i]
            diff[i] = 0
    
    for x in nums:
        if x == 0:
            update()
            
            dp.append(0)
            diff.append(0)
            
            for i in range(cur, 0, -1):
                dp[i] = fmax(dp[i], dp[i - 1])
        
        elif x > 0:
            if x <= cur:
                diff[x] += 1
                diff[cur + 1] -= 1
        
        else:
            if -x <= cur:
                diff[0] += 1
                diff[cur + x + 1] -= 1
    
    update()
    print(max(dp))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m;
	cin >> n >> m;

	vector<int> dp = {0}, diff = {0, 0};
	int cur = 0;

	auto update = [&] () -> void {
		cur ++;
		for (int i = 0; i < cur; i ++)
			diff[i + 1] += diff[i];
		for (int i = 0; i < cur; i ++)
			dp[i] += diff[i], diff[i] = 0;
	};

	while (n --) {
		int x;
		cin >> x;

		if (x == 0) {
			update();
			dp.emplace_back(0);
			diff.emplace_back(0);
			for (int i = cur; i > 0; i --)
				dp[i] = max(dp[i], dp[i - 1]);
		}
		else if (x > 0) {
			if (x <= cur) {
				diff[x] ++;
				diff[cur + 1] --;
			}
		}
		else {
			if (-x <= cur) {
				diff[0] ++;
				diff[cur + x + 1] --;
			}
		}
	}

	update();
	cout << *max_element(dp.begin(), dp.end());

	return 0;
}
```
