**提示 1：** 第 $i$ 行第 $j$ 个数变化后到哪里去了？

**提示 2：** 找到一个必要条件。尝试构造。

我们可以维护第 $i$ 行第 $j$ 个元素最后会到哪里去。

因为行列操作相互独立，同行的元素变换后也同行，同列的元素变换后也同列。所以可以设第 $i$ 行的变到 $r[i]$ 去，第 $j$ 列变到 $c[j]$ 去。

于是我们要求的就是 $A[i][j]=A[r[i]][c[j]]$ 对所有的 $i,j$ 都成立。

假设 $i$ 最少经历 $x$ 次 $i=r[i]$ 赋值操作会恢复原数值，而 $j$ 最少经历 $y$ 次 $j=c[j]$ 赋值操作会恢复原数值（也就是把排列看成一个环，看环长）。如果 $x\neq y$ ，则不妨从 $(i,j)$ 出发，将 $i,j$ 分别进行 $r,c$ 的变换 $\min(x,y)$ 次，则其中一个数值不变，另一个改变，即两个位置同行 / 同列。而变换前后数值应当相同，而同行 / 同列的元素数值应当不同，因此产生矛盾。

所以 $r$ 中每个环长等于 $c$ 中每个环长，这也可以推出 $r$ 内每个环长相等， $c$ 也一样。

所以可以将 $r$ 视为若干个相同大小的组， $c$ 也一样。

先考虑每一组之间构造。先考虑基础情况，就是两个形如 $1\to 2\to 3\to\dots\to n\to 1$ 的环如何构造，最后再映射回去就行。

而多组的情况下可以在每一对组之间分别构造，设环的长度是 $x$ ，相当于分别填入一组 $\bmod x$ 的解，最后再对不同的块加上 $0,x,2x,\dots$ ，就可以在之前保证题目的 $A[i][j]=A[r[i]][c[j]]$ 条件的基础上，进一步让行列不重复。

具体而言，我们将行列映射回来，就相当于填入有了 $\frac{n}{x}\times\frac{n}{x}$ 的分块矩阵，让每行的块和每列的块不加相同的数是很容易的。

时间复杂度为 $\mathcal{O}(n^2+q)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, t = MII()
    
    pr = list(range(n))
    pc = list(range(n))
    
    for _ in range(t):
        c, i, j = LI()
        i = int(i) - 1
        j = int(j) - 1
        
        if c == 'R': pr[i], pr[j] = pr[j], pr[i]
        else: pc[i], pc[j] = pc[j], pc[i]
    
    vis_r = [0] * n
    groups_r = []
    
    for i in range(n):
        if not vis_r[i]:
            cur = i
            group = []
            
            while not vis_r[cur]:
                vis_r[cur] = 1
                group.append(cur)
                cur = pr[cur]
            
            groups_r.append(group)
    
    vis_c = [0] * n
    groups_c = []
    
    for i in range(n):
        if not vis_c[i]:
            cur = i
            group = []
            
            while not vis_c[cur]:
                vis_c[cur] = 1
                group.append(cur)
                cur = pc[cur]
            
            groups_c.append(group)
    
    m = len(groups_r)
    k = len(groups_r[0])
    
    for x in groups_r:
        if len(x) != k:
            exit(print('*'))
    
    for x in groups_c:
        if len(x) != k:
            exit(print('*'))
    
    grid = [[0] * n for _ in range(n)]
    
    for i in range(m):
        for j in range(m):
            start = (i + j) % m * k + 1
            
            for vi in range(k):
                for vj in range(k):
                    grid[groups_r[i][vi]][groups_c[j][vj]] = start + (vi - vj) % k
    
    print('\n'.join(' '.join(map(str, x)) for x in grid))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, t;
	cin >> n >> t;

	vector<int> pr(n), pc(n);

	iota(pr.begin(), pr.end(), 0);
	iota(pc.begin(), pc.end(), 0);

	while (t --) {
		char c; int i, j;
		cin >> c >> i >> j;
		i --, j --;
		if (c == 'R') swap(pr[i], pr[j]);
		else swap(pc[i], pc[j]);
	}

	vector<int> vis_r(n, 0), vis_c(n, 0);
	vector<vector<int>> groups_r, groups_c;

	for (int i = 0; i < n; i ++) {
		if (!vis_r[i]) {
			int cur = i;
			vector<int> group;

			while (!vis_r[cur]) {
				vis_r[cur] = 1;
				group.emplace_back(cur);
				cur = pr[cur];
			}

			groups_r.emplace_back(group);
		}
	}

	for (int i = 0; i < n; i ++) {
		if (!vis_c[i]) {
			int cur = i;
			vector<int> group;

			while (!vis_c[cur]) {
				vis_c[cur] = 1;
				group.emplace_back(cur);
				cur = pc[cur];
			}

			groups_c.emplace_back(group);
		}
	}

	int m = groups_r.size(), k = groups_r[0].size();

	for (auto &x: groups_r) {
		if (x.size() != k) {
			return cout << '*', 0;
		}
	}

	for (auto &x: groups_c) {
		if (x.size() != k) {
			return cout << '*', 0;
		}
	}

	vector<vector<int>> ans(n, vector<int>(n));

	for (int i = 0; i < m; i ++) {
		for (int j = 0; j < m; j ++) {
			int start = (i + j) % m * k + 1;

			for (int di = 0; di < k; di ++) {
				for (int dj = 0; dj < k; dj ++) {
					ans[groups_r[i][di]][groups_c[j][dj]] = start + (di - dj + k) % k;
				}
			}
		}
	}

	for (int i = 0; i < n; i ++) {
		for (int j = 0; j < n; j ++) {
			cout << ans[i][j] << " \n"[j == n - 1];
		}
	}

	return 0;
}
```
