**提示 1：** 首先，先求出单个元素博弈情况下的 SG 函数。

**提示 2：** 我们最后求的跟这个 SG 函数有什么关系？

如果你不熟悉博弈论的 SG 函数，请查阅 OI-wiki 的相关资料，这里不做过多解释。

这题里，我们先求单个元素博弈的 SG 函数。设 $x$ 大小对应的 SG 函数数值是 $sg(x)$ 。则我们要求的就是 $[x-nums[x],x-1]$ 的位置对应的 MEX 数值。

从小到大考虑各个 SG 函数。因为这里转移的区间是连续且靠近要求的那个 $x$ 的，所以我们只在乎每个 SG 数值对应的最靠后的位置。我们可以用线段树维护它，这样我们只需二分找到第一个不在 $[x-nums[x],x-1]$ 区间内的元素的数值即可。我们线段树维护前缀最小出现位置。（用树状数组也可以）

接下来我们最终的数组需要满足什么条件呢？

首先，其本身 SG 函数的数值的异或和得等于 $0$ （如果这个不知道你也需要在 OI-wiki 中重新学习 SG 函数）。这个怎么求呢？我们相当于有一个序列，反复取 $r$ 轮，要求最后异或和等于 $0$ 的方案数。这件事我们可以用 FWT，快速幂后再 IFWT 即可。这个不会的话可以去学习快速沃尔什变换。

那么我们多算了哪些情况呢？多算了那些 “不存在异或值不等于 $0$ 的状态，使得将其中一个元素减去 $K$ 后等于当前状态”。

所以结果数组中的每个元素，要么不可能是由减 $K$ 得到的，要么减去 $K$ 前和当前状态的 SG 函数结果一致。否则将该位置 $+K$ 就是一个合法的异或值不等于 $0$ 的初始状态，不应排除。

于是取出这个子集再进行上述的 FWT 过程即可。

时间复杂度为 $\mathcal{O}(n\log n+n\log r)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():  
    mod = 998244353
    rev2 = (mod + 1) // 2
    
    def fwt_xor(nums, bit, type):
        for i in range(bit):
            mask = 1 << i
            for j in range(0, 1 << bit, 2 * mask):
                for k in range(mask):
                    x, y = nums[j + k], nums[j + mask + k]
                    nums[j + k] = x + y
                    nums[j + mask + k] = x - y
                    if type == -1:
                        nums[j + k] = nums[j + k] * rev2 % mod
                        nums[j + mask + k] = nums[j + mask + k] * rev2 % mod
    
    n, k, r = MII()
    nums = LII()
    
    sg = [0] * (n + 1)
    seg = SegTree(fmin, n + 1, [-1] * (n + 1))
    
    seg.set(0, 0)
    
    for i in range(1, n + 1):
        sg[i] = seg.max_right(0, lambda x: x >= i - nums[i - 1])
        seg.set(sg[i], i)
    
    ans = 0
    
    cnt = [0] * (1 << 19)
    
    for x in sg:
        cnt[x] += 1
    
    fwt_xor(cnt, 19, 1)
    
    for i in range(1 << 19):
        cnt[i] = pow(cnt[i], r, mod)
    
    fwt_xor(cnt, 19, -1)
    
    ans += cnt[0]
    
    cnt = [0] * (1 << 19)
    
    for i in range(n + 1):
        if i + k > n or sg[i] == sg[i + k]:
            cnt[sg[i]] += 1
    
    fwt_xor(cnt, 19, 1)
    
    for i in range(1 << 19):
        cnt[i] = pow(cnt[i], r, mod)
    
    fwt_xor(cnt, 19, -1)
    
    ans -= cnt[0]
    
    print(ans % mod)
```

C++ 做法如下——

```cpp []
const int mod = 998244353, rev2 = (mod + 1) / 2;

int op(int x, int y) {return min(x, y);}

int e() {return 1'000'000;}

long long quickPow(long long base, long long power, long long mod) {
	if (power == 0) return 1 % mod;
	long long cur = quickPow(base, power / 2, mod);
	return power & 1 ? base * cur % mod * cur % mod : cur * cur % mod; 
}

void fwt_xor(vector<int>& nums, int bit, int type) {
	for (int i = 0; i < bit; i++) {
		int mask = 1 << i;
		for (int j = 0; j < (1 << bit); j += 2 * mask) {
			for (int k = 0; k < mask; k ++) {
				int x = nums[j + k], y = nums[j + mask + k];
				nums[j + k] = (x + y) % mod;
				nums[j + mask + k] = (x - y + mod) % mod;
				if (type == -1) {
					nums[j + k] = 1ll * nums[j + k] * rev2 % mod;
					nums[j + mask + k] = 1ll * nums[j + mask + k] * rev2 % mod;
				}
			}
		}
	}
}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, k, r;
	cin >> n >> k >> r;

	vector<int> sg(n + 1, -1);
	atcoder::segtree<int, op, e> seg(sg);

	sg[0] = 0; seg.set(0, 0);

	for (int i = 1; i <= n; i ++) {
		int v;
		cin >> v;
		sg[i] = seg.max_right(0, [&] (int x) {return x >= i - v;});
		seg.set(sg[i], i);
	}

	int ans = 0;

	vector<int> cnt(1 << 19, 0);

	for (auto &x: sg) cnt[x] ++;

	fwt_xor(cnt, 19, 1);

	for (int i = 0; i < 1 << 19; i ++) cnt[i] = quickPow(cnt[i], r, mod);

	fwt_xor(cnt, 19, -1);

	ans += cnt[0];

	fill(cnt.begin(), cnt.end(), 0);

	for (int i = 0; i <= n; i ++) {
		if (i + k > n || sg[i] == sg[i + k]) {
			cnt[sg[i]] ++;
		}
	}

	fwt_xor(cnt, 19, 1);

	for (int i = 0; i < 1 << 19; i ++) cnt[i] = quickPow(cnt[i], r, mod);

	fwt_xor(cnt, 19, -1);

	ans -= cnt[0];

	cout << (ans + mod) % mod;

	return 0;
}
```
