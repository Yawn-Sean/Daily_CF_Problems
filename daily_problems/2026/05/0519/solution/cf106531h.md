**提示 1：** 如果有环，肯定不行。

**提示 2：** 如果没有环，有拓扑序，DP 。

如果存在环形的依赖关系，不可能完成。需要用拓扑排序判断。

否则，整个图存在拓扑序，因此执行拓扑排序的过程。对于 $u$ 课程而言，最早完成时间如何计算呢？

首先，至少等于前置课程的最早完成时间的最大值加一。

而接下来就是避开那些不让修课的时间。这里可以用哈希表判当前时间是否出现了，一旦出现不断 $+1$ 即可。

于是用上述 DP 可以求解问题。

时间复杂度为 $\mathcal{O}(n+m+k)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():  
    n, m = MII()
    
    path = [[] for _ in range(n)]
    indeg = [0] * n
    
    for _ in range(m):
        u, v = GMI()
        path[u].append(v)
        indeg[v] += 1
    
    k = II()
    invalid_notes = [[] for _ in range(n)]
    
    for i in range(k):
        j, x = MII()
        invalid_notes[x - 1].append(j)
    
    dp = [1] * n
    
    vis = [0] * n
    stk = [i for i in range(n) if indeg[i] == 0]
    
    while stk:
        u = stk.pop()
        vis[u] = 1
        
        invalid_notes[u].sort()
        for x in invalid_notes[u]:
            if dp[u] == x:
                dp[u] += 1
        
        for v in path[u]:
            indeg[v] -= 1
            dp[v] = fmax(dp[v], dp[u] + 1)
            
            if indeg[v] == 0:
                stk.append(v)
    
    print(max(dp) if sum(vis) == n else -1)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m;
	cin >> n >> m;

	vector<vector<int>> path(n);
	vector<int> indeg(n, 0);

	while (m --) {
		int u, v;
		cin >> u >> v;
		u --, v --;
		path[u].emplace_back(v);
		indeg[v] ++;
	}

	int k;
	cin >> k;

	vector<vector<int>> invalids(n);

	while (k --) {
		int j, x;
		cin >> j >> x;
		invalids[x - 1].emplace_back(j);
	}

	vector<int> dp(n, 1), stk, vis(n, 0);

	for (int i = 0; i < n; i ++) {
		if (!indeg[i]) {
			stk.emplace_back(i);
		}
	}

	while (!stk.empty()) {
		int u = stk.back(); stk.pop_back();
		vis[u] = 1;

		sort(invalids[u].begin(), invalids[u].end());

		for (auto &x: invalids[u]) {
			if (dp[u] == x) {
				dp[u] ++;
			}
		}

		for (auto &v: path[u]) {
			dp[v] = max(dp[v], dp[u] + 1);
			indeg[v] --;
			if (!indeg[v]) stk.emplace_back(v);
		}
	}

	if (*min_element(vis.begin(), vis.end())) cout << *max_element(dp.begin(), dp.end());
	else cout << -1;

	return 0;
}
```
