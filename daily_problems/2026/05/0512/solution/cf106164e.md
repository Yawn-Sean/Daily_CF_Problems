**提示 1：** 单次询问怎么做？预处理什么？预处理的东西怎么用？

对于每次查询，我们可以二分找答案。在所有的边权中进行二分，提取出 $p$ 不超过对应数值的边，再计算任意两点之间的用时。

检查合法性，第一类查询只需看是否 $u$ 到每个点的距离都不超过 $h$ ，第二类查询只需看 $i=1,\dots,n$ 中是否存在一个点到每个点的距离都不超过 $h$ 。

所以总结起来，要预处理 $p$ 小于等于某个数值的边形成的图中，两点之间的距离。

因此考虑先将边关于 $p$ 排序，这样我们的图选的就是一个前缀。

一个个添加边，在添加边的过程中，可以利用这条边进行最短路的更新。因为只有走过这条边的最短路可能改变，因此用 $dis[i][u]+h+dis[v][j]$ 更新 $dis[i][j]$ 即可。这里枚举所有的 $i,j$ 。

而对于每个点而言，重要信息就是其到任意一个点的最大距离，所以存下这些数据。

这样，对于第一类查询，也就直接在这个数组上二分，找到最早使得总距离小于等于 $h$ 的边的位置即可。

第二类查询，我们相当于要对所有的 $i$ 取最大距离的最小值，这件事也很容易预处理，因此也可以二分找到那条对应的边。也就确定了 $p$ 的答案。

这里有一个坑。我们要考虑的不是到这个位置的所有的边，而是 $p$ 不超过这个位置的所有的边。因为后面可能还有一些新增的等级为 $p$ 的边，这些边可能使得其他起点也称为可行的。

而我们要找到最小的满足条件的起点。当然朴素的做法就是枚举 $1,2,\dots, n$ ，是可以通过的。你也可以进一步优化。这个最值问题也可以考虑二分，只要我们预处理到第 $j$ 条边的情况下 $1\sim i$ 对应的到其他点的最大距离的最小值，这样就满足可以二分的性质了。

时间复杂度为 $\mathcal{O}(mn^2+q(\log n +\log m))$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    n, m, q = MII()
    edges = []
    
    for _ in range(m):
        u, v, p, h = MII()
        u -= 1
        v -= 1
        edges.append((u, v, p, h))
    
    edges.sort(key=lambda x: x[2])
    
    inf = 10 ** 13
    
    dis = [[inf] * n for _ in range(n)]
    
    for i in range(n):
        dis[i][i] = 0
    
    saved = [[] for _ in range(n)]
    
    for i, (u, v, p, h) in enumerate(edges):
        for x in range(n):
            for y in range(n):
                dis[x][y] = fmin(dis[x][y], dis[x][u] + dis[v][y] + h)
        
        for x in range(n):
            saved[x].append(max(dis[x]))
    
    edges.reverse()
    
    for val in saved:
        val.reverse()
    
    min_saved = [val[:] for val in saved]
    
    for i in range(1, n):
        for j in range(m):
            min_saved[i][j] = fmin(min_saved[i][j], min_saved[i - 1][j])
    
    outs = []
    
    for _ in range(q):
        query = LII()
        
        if query[0] == 1:
            u = query[1] - 1
            h = query[2]
            
            l, r = 0, m - 1
            
            while l <= r:
                mid = (l + r) // 2
                
                if saved[u][mid] > h: r = mid - 1
                else: l = mid + 1
            
            if r < 0: outs.append('-1')
            else: outs.append(str(edges[r][2]))
        
        else:
            h = query[1]
            
            l, r = 0, m - 1
                
            while l <= r:
                mid = (l + r) // 2
                
                if min_saved[n - 1][mid] > h: r = mid - 1
                else: l = mid + 1
            
            if r < 0: outs.append('-1 -1')
            else:
                saved_r = r
                l = 0
                
                while l <= r:
                    mid = (l + r) // 2
                    
                    if edges[mid][2] == edges[saved_r][2]: r = mid - 1
                    else: l = mid + 1
                
                pos = l
                
                l, r = 0, n - 1
                
                while l <= r:
                    mid = (l + r) // 2
                    
                    if min_saved[mid][pos] <= h: r = mid - 1
                    else: l = mid + 1
                
                outs.append(f'{l + 1} {edges[pos][2]}')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, m, q;
	cin >> n >> m >> q;

	vector<array<int, 4>> edges(m);
	for (auto &[u, v, p, h]: edges) {
		cin >> u >> v >> p >> h;
		u --, v --;
	}

	sort(edges.begin(), edges.end(), [&] (array<int, 4> &x, array<int, 4> &y) {return x[2] < y[2];});

	long long inf = 1e13;

	vector<vector<long long>> dis(n, vector<long long>(n, inf)), saved(n);

	for (int i = 0; i < n; i ++) dis[i][i] = 0;

	for (auto &[u, v, p, h]: edges) {
		for (int i = 0; i < n; i ++) {
			for (int j = 0; j < n; j ++) {
				dis[i][j] = min(dis[i][j], dis[i][u] + h + dis[v][j]);
			}
		}

		for (int i = 0; i < n; i ++) {
			saved[i].emplace_back(*max_element(dis[i].begin(), dis[i].end()));
		}
	}

	for (auto &x: saved) reverse(x.begin(), x.end());

	reverse(edges.begin(), edges.end());

	auto min_saved = saved;

	for (int i = 1; i < n; i ++) {
		for (int j = 0; j < m; j ++) {
			min_saved[i][j] = min(min_saved[i][j], min_saved[i - 1][j]);
		}
	}

	while (q --) {
		int t;
		cin >> t;

		if (t == 1) {
			int u;
			long long h;
			cin >> u >> h;
			u --;

			int l = 0, r = m - 1;

			while (l <= r) {
				int mid = (l + r) / 2;

				if (saved[u][mid] > h) r = mid - 1;
				else l = mid + 1;
			}

			if (r < 0) cout << -1 << '\n';
			else cout << edges[r][2] << '\n';
		}
		else {
			long long h;
			cin >> h;

			int l, r;

			l = 0, r = m - 1;

			while (l <= r) {
				int mid = (l + r) / 2;

				if (min_saved[n - 1][mid] > h) r = mid - 1;
				else l = mid + 1;
			}

			if (r < 0) cout << -1 << ' ' << -1 << '\n';
			else {
				int saved_r = r;
				l = 0;

				while (l <= r) {
					int mid = (l + r) / 2;
					
					if (edges[mid][2] == edges[saved_r][2]) r = mid - 1;
					else l = mid + 1; 
				}

				int pos = l;

				l = 0, r = n - 1;

				while (l <= r) {
					int mid = (l + r) / 2;

					if (min_saved[mid][pos] <= h) r = mid - 1;
					else l = mid + 1;
				}

				cout << l + 1 << ' ' << edges[pos][2] << '\n';
			}
		}
	}

	return 0;
}
```
