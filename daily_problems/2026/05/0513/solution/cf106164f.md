**提示 2：** 如何在每个答案下求对应的开心值。

首先，选取的 threshold 可能是什么呢？我们选取了一个 threshold 后，相当于取的是大于这个边界的所有商店进行考虑，于是如果选择商店的集合相同，那么对应的开心值的最终结果也相同。

所以我们只需考虑哪些 threshold 会带来不同的结果。发现就是 $0$ 或者是数组中出现过的数字。

我们从大到小考虑 threshold ，这样就相当于不断新增商店，维护 $p$ 数值对应最后的开心值。

接下来怎么确定这个数值呢？我们先二分找到可以开始进入的商店，接下来连续的一段商店都可行。一段结束后再二分找下一段。

因为要修改数组的时候，找某个起点后数值之和不超过 $v$ 的位置，所以可以使用线段树二分。

为什么这样做复杂度是对的呢？每一轮这样做的复杂度都是 $\mathcal{O}(\log n)$ 的。而每一轮结束后， $p$ 的数值至少减半，因为要么到数组结尾了，要么变成了比最后一次减去的数还要小的数，也就是剩下的数比减去的最后一个数，也就比减去的数之和要小，也就至少减半了。

所以总共最多 $\mathcal{O}(\log p)$ 轮，复杂度有保证。

时间复杂度为 $\mathcal{O}(n\log n\log p)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main(): 
    bound = 10 ** 18 + 5
    def add(x, y):
        return fmin(x + y, bound)
    
    t = II()
    outs = []
    
    for _ in range(t):
        n, p = MII()
        
        hs = []
        ps = []
        
        for _ in range(n):
            h_val, p_val = MII()
            hs.append(h_val)
            ps.append(p_val)
        
        order = sorted(range(n), key=lambda x: -hs[x])
        
        seg = SegTree(add, 0, n)
        seg_happiness = SegTree(add, 0, n)
        
        ans = -1
        chosen_threshold = -1
        
        for i in range(n):
            seg.set(order[i], ps[order[i]])
            seg_happiness.set(order[i], hs[order[i]])
            
            if i == n - 1 or hs[order[i]] != hs[order[i + 1]]:
                cur_pos = 0
                cur_p = p
                
                res = 0
                
                while True:
                    l, r = cur_pos, n - 1
                    while l <= r:
                        mid = (l + r) // 2
                        if ps[mid] > cur_p: l = mid + 1
                        else: r = mid - 1
                    
                    cur_pos = l
                    if cur_pos == n: break
                    
                    n_cur_pos = seg.max_right(cur_pos, lambda x: x <= cur_p)
                    cur_p -= seg.prod(cur_pos, n_cur_pos)
                    res += seg_happiness.prod(cur_pos, n_cur_pos)
                    
                    cur_pos = n_cur_pos
                
                if res >= ans:
                    ans = res
                    chosen_threshold = hs[order[i + 1]] if i != n - 1 else 0
        
        outs.append(f'{ans} {chosen_threshold}')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
long long bound = 2e18;

long long op(long long x, long long y) {
	return min(x + y, bound);
}

long long e() {return 0;}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t;
	cin >> t;

	while (t --) {
		int n;
		long long p;
		cin >> n >> p;

		vector<pair<long long, long long>> stalls(n);
		for (auto &[x, y]: stalls) cin >> x >> y;

		vector<int> order(n);
		iota(order.begin(), order.end(), 0);
		sort(order.begin(), order.end(), [&] (int i, int j) {return stalls[i].first > stalls[j].first;});

		long long ans = -1;
		int chosen_threshold = -1;

		atcoder::segtree<long long, op, e> seg(n), seg_happiness(n);

		for (int i = 0; i < n; i ++) {
			seg.set(order[i], stalls[order[i]].second);
			seg_happiness.set(order[i], stalls[order[i]].first);

			if (i == n - 1 || stalls[order[i]].first != stalls[order[i + 1]].first) {
				int cur_pos = 0;
				long long cur_p = p, res = 0;

				while (true) {
					int l = cur_pos, r = n - 1;

					while (l <= r) {
						int mid = (l + r) / 2;
						if (stalls[mid].second > cur_p) l = mid + 1;
						else r = mid - 1;
					}

					cur_pos = l;
					if (cur_pos == n) break;

					int n_cur_pos = seg.max_right(cur_pos, [&] (long long x) {return x <= cur_p;});
					cur_p -= seg.prod(cur_pos, n_cur_pos);
					res += seg_happiness.prod(cur_pos, n_cur_pos);

					cur_pos = n_cur_pos;
				}

				if (res >= ans) {
					ans = res;
					chosen_threshold = (i != n - 1 ? stalls[order[i + 1]].first : 0);
				}
			}
		}

		cout << ans << ' ' << chosen_threshold << '\n';
	}

	return 0;
}
```
