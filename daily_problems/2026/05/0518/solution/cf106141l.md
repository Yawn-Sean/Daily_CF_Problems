**提示 1：**

时间复杂度为 $\mathcal{O}(n\log n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():  
    n = II()
    nums = LII()
    
    vals = [1 if x >= 0 else -1 for x in nums]
    st_range = sorted(range(n), key=lambda x: -abs(nums[x]))
    
    total = sum(abs(x) for x in nums)
    val = n
    
    ma_total = total
    ma_val = n
    chosen = 0
    
    for i in range(n):
        total += 2 * abs(nums[st_range[i]])
        val += 8
        
        if total * total * ma_val > ma_total * ma_total * val:
            ma_total = total
            ma_val = val
            chosen = i + 1
    
    for i in range(chosen):
        vals[st_range[i]] *= 3
    
    print(*vals)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e5 + 5;
	vector<int> pr(M, 1);
	pr[0] = 0, pr[1] = 0;

	for (int i = 2; i < M; i ++) {
		if (pr[i]) {
			for (int j = 2 * i; j < M; j += i) {
				pr[j] = 0;
			}
		}
	}

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;

		vector<int> c1(n + 1, 0), c2(n + 1, 0);

		for (int i = 0; i < n; i ++) {
			int x; cin >> x;
			c1[x] ++;
		}

		for (int i = 0; i < n; i ++) {
			int x; cin >> x;
			c2[x] ++;
		}

		for (int i = 1; i <= n; i ++) {
			for (int j = 2 * i; j <= n; j += i) {
				c1[i] += c1[j];
				c2[i] += c2[j];
			}
		}

		vector<long long> cnt(n + 1, 0);
		for (int i = 0; i <= n; i ++) {
			cnt[i] = 1ll * c1[i] * c2[i];
		}

		for (int i = n; i >= 1; i --) {
			for (int j = 2 * i; j <= n; j += i) {
				cnt[i] -= cnt[j];
			}
		}

		long long ans = 0;
		for (int i = 0; i <= n; i ++) {
			if (pr[i]) {
				ans += cnt[i];
			}
		}

		cout << ans << '\n';
	}

	return 0;
}
```
