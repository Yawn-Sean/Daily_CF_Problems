**提示 1：** 考虑 gcd 是 $x$ 的倍数的数对，这个怎么求？

**提示 2：** 如何求 gcd 恰好是 $x$ 的数对个数。

先看提示 1 。我们可以统计每个数组中 $x$ 的倍数的个数（直接枚举 $x,2x,\dots$ 的出现个数，复杂度是 $\mathcal{O}(\sum n/x)=\mathcal{O}(n\log n)$ ），相乘就是 gcd 是 $x$ 的倍数的数对个数。

接下来为了得到 gcd 恰好是 $x$ 的数对的个数，我们可以用 gcd 是 $x$ 的倍数的数对的数量，减去 gcd 恰好是 $2x,3x,\dots$ 的情况的数对的数量。这里的枚举是 $\mathcal{O}(n/x)$ 的，所以整体复杂度是 $\mathcal{O}(n\log n)$ 。

最后对质数的位置的结果求和即可。

时间复杂度为 $\mathcal{O}(n\log n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():  
    M = 10 ** 5 + 5
    pr = [1] * M
    pr[0] = 0
    pr[1] = 0
    
    for i in range(2, M):
        if pr[i]:
            for j in range(2 * i, M, i):
                pr[j] = 0
    
    t = II()
    outs = []
    
    for _ in range(t):
        n = II()
        v1 = LII()
        v2 = LII()
        
        c1 = [0] * (n + 1)
        c2 = [0] * (n + 1)
        
        for x in v1:
            c1[x] += 1
        
        for x in v2:
            c2[x] += 1
        
        for i in range(1, n + 1):
            for j in range(2 * i, n + 1, i):
                c1[i] += c1[j]
                c2[i] += c2[j]
        
        c = [x * y for x, y in zip(c1, c2)]
        
        for i in range(n, 0, -1):
            for j in range(2 * i, n + 1, i):
                c[i] -= c[j]
        
        ans = 0
        for i in range(n + 1):
            if pr[i]:
                ans += c[i]
        
        outs.append(ans)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e5 + 5;
	vector<int> pr(M, 1);
	pr[0] = 0, pr[1] = 0;

	for (int i = 2; i < M; i ++) {
		if (pr[i]) {
			for (int j = 2 * i; j < M; j += i) {
				pr[j] = 0;
			}
		}
	}

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;

		vector<int> c1(n + 1, 0), c2(n + 1, 0);

		for (int i = 0; i < n; i ++) {
			int x; cin >> x;
			c1[x] ++;
		}

		for (int i = 0; i < n; i ++) {
			int x; cin >> x;
			c2[x] ++;
		}

		for (int i = 1; i <= n; i ++) {
			for (int j = 2 * i; j <= n; j += i) {
				c1[i] += c1[j];
				c2[i] += c2[j];
			}
		}

		vector<long long> cnt(n + 1, 0);
		for (int i = 0; i <= n; i ++) {
			cnt[i] = 1ll * c1[i] * c2[i];
		}

		for (int i = n; i >= 1; i --) {
			for (int j = 2 * i; j <= n; j += i) {
				cnt[i] -= cnt[j];
			}
		}

		long long ans = 0;
		for (int i = 0; i <= n; i ++) {
			if (pr[i]) {
				ans += cnt[i];
			}
		}

		cout << ans << '\n';
	}

	return 0;
}
```
