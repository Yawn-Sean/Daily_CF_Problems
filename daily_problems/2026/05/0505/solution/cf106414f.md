**提示 1：** 考虑中间的分割点，前面选一个数，后一部分选另一个数。

**提示 2：** 存哪些备选。

因为整体有两段，一段是一个数值，所以考虑枚举中间的分割位置。也就是拆分为前缀和后缀，分别选取一个相同的数。

我们可以统计每个前缀 / 后缀中每个元素的出现频率。

首先第一直觉是直接选择出现频率最高的数出现的次数。

但左右两端对应数可能数值相同，这在本题中是不允许的，所以考虑维护前两名。这样就一定能找到最优组合了。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    t = II()
    outs = []
    
    for _ in range(t):
        n, m = MII()
        nums = LII()
        
        order = sorted(range(n), key=lambda x: nums[x])
        
        ans = []
        bound = n
        
        for i in range(2, n):
            if nums[order[i]] >= m:
                if nums[order[0]] + nums[order[1]] + nums[order[i]] <= 2 * m:
                    ans = [order[0], order[1], order[i]]
                bound = i
                break
        
        if not ans and bound >= 3:
            for i in range(4):
                tmp = []
                for j in range(i):
                    tmp.append(order[j])
                for j in range(3 - i):
                    tmp.append(order[bound - 1 - j])
                
                total = 0
                for x in tmp:
                    total += nums[x]
                
                if m <= total <= 2 * m:
                    ans = tmp
                    break
        
        if ans: outs.append(' '.join(str(x + 1) for x in ans))
        else: outs.append('-1')
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int t;
	cin >> t;

	while (t --) {
		int n, m;
		cin >> n >> m;

		vector<int> nums(n);
		for (auto &x: nums) cin >> x;

		vector<int> order(n);
		iota(order.begin(), order.end(), 0);
		sort(order.begin(), order.end(), [&] (int i, int j) {return nums[i] < nums[j];});

		int bound = n;
		vector<int> ans;

		for (int i = 2; i < n; i ++) {
			if (nums[order[i]] >= m) {
				if (nums[order[0]] + nums[order[1]] + nums[order[i]] <= 2 * m)
					ans = {order[0], order[1], order[i]};
				bound = i;
				break;
			}
		}
		
		if (ans.empty() && bound >= 3) {
			for (int i = 0; i <= 3; i ++) {
				vector<int> tmp;
				for (int j = 0; j < i; j ++) tmp.emplace_back(order[j]);
				for (int j = 0; j < 3 - i; j ++) tmp.emplace_back(order[bound - 1 - j]);

				int total = 0;
				for (auto &x: tmp) total += nums[x];

				if (total >= m && total <= 2 * m) {
					ans = tmp;
					break;
				}
			}
		}

		if (ans.empty()) cout << -1 << '\n';
		else cout << ans[0] + 1 << ' ' << ans[1] + 1 << ' ' << ans[2] + 1 << '\n';
	}

	return 0;
}
```
