**提示 1：** 总错排数量很多，所以 $n$ 很大的时候……

**提示 2：** 如何一位位确定答案？

首先，错排大概量级是 $n!$ 的，所以 $k$ 相比而言实在是很小。

所以在 $n$ 很大的时候，前两项一定是 $2,1$ ，可以将问题转化为 $n-2$ 的问题。这个过程可以执行，直到 $n\leq 19$ 。

接下来就可以一位位确定答案了。

到某一步的时候剩余的元素有多少种凑法怎么确定呢？我们先看可能的状态。

假设剩余 $x$ 个位置没填数，而也就有 $x$ 个数没有被填。这两个东西的交集大小为 $y$ ，这些状态就足够计算剩余元素的填法了（可以向下为啥）。

这个 $dp[x][y]$ 怎么求呢？我们选取 $y$ 中的最小元素 $p$，看对应位置填什么元素：

- 如果填的数 $v$ 也在交集中，则 $x$ 变为 $x-1$ ， $y$ 变为 $y-2$ ，因为 $p,v$ 都不再是交集内的元素了。而选择方式共有 $y-1$ 中，因为不能选自己。

- 否则， $x$ 变为 $x-1$ ， $y$ 变为 $y-1$ ，因为就减少了 $p$ 这么一个元素。选择方式有 $x-y$ 种，可选集大小减去交集大小即可。

因此 $dp[x][y]=dp[x-1][y-2]\times(y-1)+dp[x-1][y-1]\times(x-y)$ 。

注意 $y=0$ 的情况下，就是任意填，初值使用 $dp[x][0]=x!$ 即可。

而假设我们填到第 $i$ 个位置，要求的是满足前缀为当前填的情况的第 $k$ 个序列，怎么办呢？

我们从小到大枚举这个位置填的数值，看剩余位置填的方案数，如果小于 $k$ ，则这个位置不能填更大的数值了，否则标号已经大于 $k$ 了，所以这个位置就填这个元素；否则，这个位置一定填更大的数，继续往后枚举。

下面代码实现中的 DP 定义有所不同。

时间复杂度为 $\mathcal{O}(n+M^2)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    dp = [[0] * 20 for _ in range(20)]
    dp[0][0] = 1
    
    for i in range(1, 20):
        dp[0][i] = i * dp[0][i - 1]
    
    for i in range(1, 20):
        for j in range(20):
            if i + j <= 19:
                dp[i][j] = dp[i - 1][j] * j
                if i >= 2:
                    dp[i][j] += dp[i - 2][j + 1] * (i - 1)
    
    t = II()
    outs = []
    
    for _ in range(t):
        n, k = MII()
        k -= 1
        
        v = fmin(n, n % 2 + 18)
        used = [0] * v
        
        res = [0] * v
        
        x, y = v, 0
        for i in range(v):
            for j in range(v):
                if used[j] or j == i: continue
                
                if used[i] and j < i:
                    nx, ny = x, y - 1
                elif used[i] or j < i:
                    nx, ny = x - 1, y
                else:
                    nx, ny = x - 2, y + 1
                
                if dp[nx][ny] <= k:
                    k -= dp[nx][ny]
                else:
                    x = nx
                    y = ny
                    used[j] = 1
                    res[i] = j
                    break
        
        ans = [0] * n
        
        for i in range(n // 2):
            ans[i * 2] = i * 2 + 1
            ans[i * 2 + 1] = i * 2
        
        for i in range(v):
            ans[n - v + i] = n - v + res[i]
        
        outs.append(' '.join(str(x + 1) for x in ans))
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	vector<vector<long long>> dp(20, vector<long long>(20, 0));
	dp[0][0] = 1;

	for (int i = 1; i < 20; i ++) dp[0][i] = i * dp[0][i - 1];

	for (int i = 1; i < 20; i ++) {
		for (int j = 0; j < 20; j ++) {
			if (i + j < 20) {
				dp[i][j] = dp[i - 1][j] * j;
				if (i >= 2) dp[i][j] += dp[i - 2][j + 1] * (i - 1);
			}
		}
	}

	int t;
	cin >> t;

	while (t --) {
		int n;
		long long k;
		cin >> n >> k;

		k --;

		int v = min(n, n % 2 + 18);
		vector<int> used(v, 0), res(v, 0);

		int x = v, y = 0;

		for (int i = 0; i < v; i ++) {
			for (int j = 0; j < v; j ++) {
				if (used[j] || j == i) continue;

				int nx, ny;
				if (used[i] && j < i) nx = x, ny = y - 1;
				else if (used[i] || j < i) nx = x - 1, ny = y;
				else nx = x - 2, ny = y + 1;

				if (dp[nx][ny] <= k) k -= dp[nx][ny];
				else {
					x = nx;
					y = ny;
					used[j] = 1;
					res[i] = j;
					break;
				}
			}
		}

		for (int i = 0; i < (n - v) / 2; i ++) cout << 2 * i + 2 << ' ' << 2 * i + 1 << ' ';
		for (int i = 0; i < v; i ++) cout << res[i] + n - v + 1 << " \n"[i == v - 1];
	}

	return 0;
}
```
