**提示 2：** DP 状态？

先看提示 1 。

首先，我们最多选择两个奇数，因为任意两个奇数异或结果只能是偶数，而偶质数只有 $2$ ，所以要让奇数之间的异或和两两为 $2$ ，最多只能选两个。

同理也至多选两个偶数。

这样我们总共最多选 $4$ 个数。

但似乎状态还是太多了，但是刚刚我们也提到了，选两个奇数的时候，我们只可能让它们异或和是 $2$ ，所以只可能是不选或者 $x$ 或者 $x,x+2$ 的形式，偶数也是一样。

于是对于一个固定的前缀，可行的奇数选取方案是 $\mathcal{O}(n)$ 的，偶数选取方案是 $\mathcal{O}(n)$ 的，因此总的选取方案数是 $\mathcal{O}(n^2)$ 。

接下来考虑 DP ，在最开头的时候，是空集，答案为 $0$ 。

而新增一个元素后，选取的子集，复杂度是 $\mathcal{O}(2^k)$ 的（ $k$ 是选取的卡片数量），选取子集判定合法后发生状态转移，更新状态转移后的结果即可。

这里为了减少状态，可以直接对选取的元素排序。

时间复杂度为 $\mathcal{O}(n^3)$ 。

Python 做法如下——

```Python []
def main():
    M = 1 << 20
    pr = [1] * M
    
    pr[0] = 0
    pr[1] = 0
    
    for i in range(2, M):
        if pr[i]:
            for j in range(2 * i, M, i):
                pr[j] = 0
    
    inf = 10 ** 18
    
    t = II()
    outs = []
    
    for _ in range(t):
        n = II()
        nums = LII()
        
        dp = Counter()
        dp[tuple()] = 0
        
        for value in nums:
            ndp = Counter()
            
            for v in dp:
                ndp[v] = fmax(ndp[v], dp[v])
                
                vals = list(v)
                k = len(vals)
                
                for i in range(1 << k):
                    tmp = [vals[j] for j in range(k) if i >> j & 1]
                    tmp.append(value)
                    
                    l = len(tmp)
                    flg = True
                    
                    for x in range(l):
                        for y in range(x):
                            if not pr[tmp[x] ^ tmp[y]]:
                                flg = False
                    
                    if flg:
                        tmp.sort()
                        tmp = tuple(tmp)
                        ndp[tmp] = fmax(ndp[tmp], dp[v])
            
            dp = ndp
            
            for x in dp:
                dp[x] += sum(x)
        
        outs.append(max(dp.values()))
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1 << 20;

	vector<int> is_prime(M, 1);
	is_prime[0] = 0, is_prime[1] = 0;

	for (int i = 2; i < M; i ++) {
		if (is_prime[i]) {
			for (int j = i * 2; j < M; j += i) {
				is_prime[j] = 0;
			}
		}
	}

	int t;
	cin >> t;

	while (t --) {
		int n;
		cin >> n;

		map<vector<int>, long long> dp;
		dp[vector<int>()] = 0;

		while (n --) {
			int x;
			cin >> x;

			map<vector<int>, long long> ndp = dp;

			for (auto [nums, val]: dp) {
				int k = nums.size();

				for (int i = 0; i < 1 << k; i ++) {
					vector<int> tmp;

					for (int j = 0; j < k; j ++) {
						if (i >> j & 1) {
							tmp.emplace_back(nums[j]);
						}
					}

					tmp.emplace_back(x);

					bool flg = true;
					for (int a = 0; a < tmp.size(); a ++) {
						for (int b = 0; b < a; b ++) {
							if (!is_prime[tmp[a] ^ tmp[b]]) {
								flg = false;
							}
						}
					}

					if (flg) {
						sort(tmp.begin(), tmp.end());
						ndp[tmp] = max(ndp[tmp], val);
					}
				}
			}

			for (auto &[k, val]: ndp) {
				for (auto &v: k) {
					val += v;
				}
			}

			dp = ndp;
		}

		long long ans = 0;
		for (auto &[x, y]: dp) ans = max(ans, y);

		cout << ans << '\n';
	}

	return 0;
}
```
