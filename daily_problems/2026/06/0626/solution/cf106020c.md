**提示 2：** 没必要检查那么多。

假设我们选择了某条边，此时有一部分已经最大公约数为 $1$ 了，那么我们取另个部分的一个叶子，断开它上面的边，则也一定满足要求。

所以我们只要检查叶子就行，而叶子只需检查整个序列除去单点后的最大公约数即可。

似乎不太好办了。

但我们只需检查是否存在一种方式使得某部分的最大公约数是 $1$ 。

假设存在一种方式使得某部分的最大公约数是 $1$ ， 且我们检查了两个叶子，这两个叶子的其余部分的最大公约数都不小于 $1$ ，设分别为 $x,y$ ，则考虑这两个叶子外的所有点的最大公约数，它们都是 $x,y$ 的倍数。

于是，如果 $x,y$ 最大公约数不为 $1$ ，则每个点的数都是这个元素的倍数，与假设中整体最大公约数为 $1$ 矛盾。

所以 $x,y$ 互质。

那如果存在一种方式使得某部分的最大公约数是 $1$ ，且我们查询了 $k$ 个点都没找到呢？则去掉这 $k$ 个点的最大公约数都大于 $1$ 且两两互质，此时如果还有别的非这 $k$ 个点的点，则这个点对应的数同时是这些最大公约数的倍数。

而这个数不能超过 $10^{18}$ ，因此在能找到答案的情况下 $k$ 不会很大，否则就推出矛盾了。

$2\times 3\times 5\times 7\times 11\times 13\times 17\times 19\times 23\times 29\times 31\times 37\times 41\times 43\times 47\times 53\gt 10^{18}$

所以 $k$ 不会超过 $15$ ，最多只需检查 $16$ 个叶子即可。

而检查叶子是否满足，可以直接查询除了这个叶子的两个区间的最大公约数再取其最大公约数，可以用线段树维护。

这里也可以把所有叶子聚到一起。假设前 $k$ 个元素是要检查的，则剩余元素的最大公约数求完后，整体就是一个长度为 $k+1$ 的序列去掉一个元素的最大公约数求解，就可以用前后缀分解完成，可以降低复杂度。（相当于少查了很多次长区间）

时间复杂度为 $\mathcal{O}(n\log M+q(\log n+\log M))$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    nums = LII()
    
    path = [[] for _ in range(n)]
    for _ in range(n - 1):
        u, v = GMI()
        path[u].append(v)
        path[v].append(u)
    
    tmp = []
    for i in range(n):
        if len(path[i]) == 1:
            tmp.append(i)
    
    for i in range(n):
        if len(path[i]) > 1:
            tmp.append(i)
    
    vals = [0] * n
    pos = [0] * n
    
    for i in range(n):
        vals[i] = nums[tmp[i]]
        pos[tmp[i]] = i
    
    seg = SegTree(math.gcd, 0, vals)
    
    q = II()
    outs = []
    
    for _ in range(q):
        i, x = MII()
        i -= 1
        
        vals[pos[i]] = x
        seg.set(pos[i], x)
        
        k = fmin(16, n)
        val = seg.prod(k, n)
        
        pref = [0]
        for i in range(k - 1):
            pref.append(math.gcd(pref[-1], vals[i]))
        
        flg = False
        for i in range(k - 1, -1, -1):
            if len(path[tmp[i]]) == 1 and math.gcd(pref[i], val) == 1:
                flg = True
            val = math.gcd(vals[i], val)
        
        outs.append('Yes' if flg else 'No')
    
    print('\n'.join(outs))
```

C++ 做法如下——

```cpp []
long long op(long long x, long long y) {return gcd(x, y);}

long long e() {return 0;}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	vector<long long> nums(n);
	for (auto &x: nums) cin >> x;

	vector<int> deg(n, 0);
	for (int i = 0; i < n - 1; i ++) {
		int u, v;
		cin >> u >> v;
		u --, v --;
		deg[u] ++, deg[v] ++;
	}

	vector<int> tmp;
	for (int i = 0; i < n; i ++) if (deg[i] == 1) tmp.emplace_back(i);
	for (int i = 0; i < n; i ++) if (deg[i] > 1) tmp.emplace_back(i);

	vector<long long> vals(n, 0);
	vector<int> pos(n, 0);

	for (int i = 0; i < n; i ++) {
		vals[i] = nums[tmp[i]];
		pos[tmp[i]] = i;
	}

	atcoder::segtree<long long, op, e> seg(vals);

	int q;
	cin >> q;

	while (q --) {
		int idx;
		long long val;
		cin >> idx >> val; idx --;

		vals[pos[idx]] = val;
		seg.set(pos[idx], val);

		int k = min(16, n);
		long long cur = seg.prod(k, n);

		vector<long long> pref = {0};
		for (int i = 0; i < k - 1; i ++) pref.emplace_back(gcd(pref.back(), vals[i]));

		bool flg = false;
		for (int i = k - 1; i >= 0; i --) {
			if (deg[tmp[i]] == 1 && gcd(pref[i], cur) == 1) flg = true;
			cur = gcd(cur, vals[i]);
		}

		cout << (flg ? "Yes" : "No") << '\n';
	}

	return 0;
}
```
