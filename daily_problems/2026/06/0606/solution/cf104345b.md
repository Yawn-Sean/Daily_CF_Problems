**提示 2：** 怎么快速找到连通块。

我们考虑端点都是查询的点构成的所有的边。这些边造出来的连通块中的任意两点都是一组满足条件的对。

那如何找到连通块呢？枚举一个端点出发的所有边是完全不对的，会被菊花图卡掉。怎么减少枚举的点数呢？

枚举一个点，看其父节点是否在查询点集中，如果在，这条边就在考虑范围内。注意到每条边都会这样被考虑到。

而找到边后，我们可以选择用这些边建图，再 BFS / DFS ；或者直接使用并查集。当然你建图 / 用完并查集后得恢复一下。

时间复杂度为 $\mathcal{O}(n+\sum k)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    path = [[] for _ in range(n)]
    
    for _ in range(n - 1):
        u, v = GMI()
        path[u].append(v)
        path[v].append(u)
    
    parent = [-1] * n
    que = [0]
    
    for u in que:
        for v in path[u]:
            if parent[u] != v:
                parent[v] = u
                que.append(v)
    
    q = II()
    outs = []
    
    vis = [0] * n
    uf = UnionFind(n)
    
    for _ in range(q):
        _, *nodes = GMI()
        
        for x in nodes:
            vis[x] = 1
        
        for x in nodes:
            if x and vis[parent[x]]:
                uf.merge(x, parent[x])
        
        ans = 0
        
        for x in nodes:
            if uf.find(x) == x:
                sz = uf.getSize(x)
                ans += sz * (sz - 1) // 2
        
        outs.append(ans)
        
        for x in nodes:
            vis[x] = 0
            uf.reset(x)
    
    print('\n'.join(map(str, outs)))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

    int n;
    cin >> n;

    vector<vector<int>> path(n);
    for (int i = 0; i < n - 1; i ++) {
        int u, v;
        cin >> u >> v;
        u --, v --;
        path[u].emplace_back(v);
        path[v].emplace_back(u);
    }

    vector<int> parent(n, -1);

    auto dfs = [&] (auto &self, int u, int p) -> void {
        parent[u] = p;
        for (auto &v: path[u]) {
            if (v != p) {
                self(self, v, u);
            }
        }
    };

    dfs(dfs, 0, -1);

    int q;
    cin >> q;

    vector<int> vis(n, 0);
    atcoder::dsu uf(n);

    while (q --) {
        int k;
        cin >> k;

        vector<int> nodes(k);
        for (auto &x: nodes) cin >> x, x --, vis[x] = 1;

        for (auto &x: nodes) {
            if (x && vis[parent[x]]) {
                uf.merge(x, parent[x]);
            }
        }

        long long ans = 0;
        for (auto &x: nodes) {
            if (uf.leader(x) == x) {
                int sz = uf.size(x);
                ans += 1ll * sz * (sz - 1) / 2;
            }
        }

        cout << ans << '\n';

        for (auto &x: nodes) vis[x] = 0, uf.reset(x);
    }

	return 0;
}
```
