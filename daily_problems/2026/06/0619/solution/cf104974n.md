**提示 1：** 第二个条件啥意思？

**提示 2：** 第一个条件如何处理？

第二个条件就是说交换所有 $2x,2x+1$ ，逆序对数量的改变量是 $D$ 。

但实际上，除了交换的元素之间，其他逆序对不会发生改变。所以相当于 $\lfloor n/2\rfloor$ 对逆序关系反转，最后改变量是 $D$ 。

假设其中有 $k$ 个开始是逆序关系，则该变量是 $(\lfloor n/2\rfloor-k)-k$ ，这个东西需要等于 $D$ ，所以可以解出对应的 $k$ 。于是这里相当于 $\lfloor n/2\rfloor$ 对元素中选取 $k$ 对作为逆序关系。

而开头的时候，我们需要任意一组 $2x,2x+1$ 不相邻。这件事不太好直接算，可以用容斥（二项式反演），求至少 $0,1,2,\dots$ 组 $2x,2x+1$ 相邻的方案数然后间接求出答案。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n, d = MII()
    mod = 998244353
    
    f = Factorial(n, mod)
    pw2 = [1] * (n + 1)
    for i in range(n):
        pw2[i + 1] = pw2[i] * 2 % mod
    
    ans = 0
    
    for i in range(n // 2 + 1):
        ans += (1 if i % 2 == 0 else -1) * f.fac(n - i) * f.combi(n // 2, i) % mod * pw2[i] % mod
        ans %= mod
    
    ans = ans * pow(pw2[n // 2], -1, mod) % mod
    
    if (n // 2 - d) % 2: print(0)
    else: print(ans * f.combi(n // 2, (n // 2 + d) // 2) % mod)
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n, mod = 998244353;
	long long d;
	cin >> n >> d;

	vector<int> f(n + 1), g(n + 1);

	f[0] = 1;
	for (int i = 1; i <= n; i ++) f[i] = 1ll * i * f[i - 1] % mod;

	g[n] = quickPow(f[n], mod - 2, mod);
	for (int i = n; i >= 1; i --) g[i - 1] = 1ll * i * g[i] % mod;

	vector<int> pw2(n + 1);
	pw2[0] = 1;

	for (int i = 1; i <= n; i ++) pw2[i] = 2ll * pw2[i - 1] % mod;

	auto comb = [&] (int x, long long y) -> int {
		if (y > x || y < 0) return 0;
		return 1ll * f[x] * g[y] % mod * g[x - y] % mod;
	};

	int ans = 0;

	for (int i = 0; i <= n / 2; i ++) {
		int val = 1ll * f[n - i] * comb(n / 2, i) % mod * pw2[i] % mod;
		if (i & 1) ans += mod - val;
		else ans += val;
		ans %= mod;
	}

	ans = ans * quickPow(pw2[n / 2], mod - 2, mod) % mod;

	if ((n / 2 - d) & 1) cout << 0;
	else cout << 1ll * ans * comb(n / 2, (n / 2 + d) / 2) % mod;

	return 0;
}
```
